204 MFRC522::StatusCode MFRC522::PCD_CalculateCRC( byte *data,
205  byte length,
206  byte *result
207  ) {
208  PCD_WriteRegister(CommandReg, PCD_Idle); // Stop any active command.
209  PCD_WriteRegister(DivIrqReg, 0x04); // Clear the CRCIRq interrupt request bit
210  PCD_SetRegisterBitMask(FIFOLevelReg, 0x80); // FlushBuffer = 1, FIFO initialization
211  PCD_WriteRegister(FIFODataReg, length, data); // Write data to the FIFO
212  PCD_WriteRegister(CommandReg, PCD_CalcCRC); // Start the calculation
213 
214  // Wait for the CRC calculation to complete. Each iteration of the while-loop takes 17.73�s.
215  word i = 5000;
216  byte n;
217  while (1) {
218  n = PCD_ReadRegister(DivIrqReg); // DivIrqReg[7..0] bits are: Set2 reserved reserved MfinActIRq reserved CRCIRq reserved reserved
219  if (n & 0x04) { // CRCIRq bit set - calculation done
220  break;
221  }
222  if (--i == 0) { // The emergency break. We will eventually terminate on this one after 89ms. Communication with the MFRC522 might be down.
223  return STATUS_TIMEOUT;
224  }
225  }
226  PCD_WriteRegister(CommandReg, PCD_Idle); // Stop calculating CRC for new content in the FIFO.
227 
228  // Transfer the result from the registers to the result buffer
229  result[0] = PCD_ReadRegister(CRCResultRegL);
230  result[1] = PCD_ReadRegister(CRCResultRegH);
231  return STATUS_OK;
232 } // End PCD_CalculateCRC()

297 bool MFRC522::PCD_Reset() {
298  PCD_WriteRegister(CommandReg, PCD_SoftReset); // Issue the SoftReset command.
299  // The datasheet does not mention how long the SoftRest command takes to complete.
300  // But the MFRC522 might have been in soft power-down mode (triggered by bit 4 of CommandReg)
301  // Section 8.8.2 in the datasheet says the oscillator start-up time is the start up time of the crystal + 37,74�s. Let us be generous: 50ms.
302  systemClock->wait_us(50*1000); //delay(50);
303  // Wait for the PowerDown bit in CommandReg to be cleared
304  Time waitingBegin = systemClock->getTime();
305  uint32_t timeout = 500000;
306  while (PCD_ReadRegister(CommandReg) & (1<<4)) {
307  // PCD still restarting - unlikely after waiting 50ms, but better safe than sorry.
308  if (systemClock->getTime() - waitingBegin > timeout)
309  return false;
310  }
311  return true;

361 bool MFRC522::PCD_PerformSelfTest() {
362  // This follows directly the steps outlined in 16.1.1
363  // 1. Perform a soft reset.
364  PCD_Reset();
365 
366  // 2. Clear the internal buffer by writing 25 bytes of 00h
367  byte ZEROES[25] = {0x00};
368  PCD_SetRegisterBitMask(FIFOLevelReg, 0x80); // flush the FIFO buffer
369  PCD_WriteRegister(FIFODataReg, 25, ZEROES); // write 25 bytes of 00h to FIFO
370  PCD_WriteRegister(CommandReg, PCD_Mem); // transfer to internal buffer
371 
372  // 3. Enable self-test
373  PCD_WriteRegister(AutoTestReg, 0x09);
374 
375  // 4. Write 00h to FIFO buffer
376  PCD_WriteRegister(FIFODataReg, 0x00);
377 
378  // 5. Start self-test by issuing the CalcCRC command
379  PCD_WriteRegister(CommandReg, PCD_CalcCRC);
380 
381  // 6. Wait for self-test to complete
382  word i;
383  byte n;
384  for (i = 0; i < 0xFF; i++) {
385  n = PCD_ReadRegister(DivIrqReg); // DivIrqReg[7..0] bits are: Set2 reserved reserved MfinActIRq reserved CRCIRq reserved reserved
386  if (n & 0x04) { // CRCIRq bit set - calculation done
387  break;
388  }
389  }
390  PCD_WriteRegister(CommandReg, PCD_Idle); // Stop calculating CRC for new content in the FIFO.
391 
392  // 7. Read out resulting 64 bytes from the FIFO buffer.
393  byte result[64];
394  PCD_ReadRegister(FIFODataReg, 64, result, 0);
395 
396  // Auto self-test done
397  // Reset AutoTestReg register to be 0 again. Required for normal operation.
398  PCD_WriteRegister(AutoTestReg, 0x00);
399 
400  // Determine firmware version (see section 9.3.4.8 in spec)
401  byte version = PCD_ReadRegister(VersionReg);
402 
403  // Pick the appropriate reference values
404  const byte *reference;
405  switch (version) {
406  case 0x88: // Fudan Semiconductor FM17522 clone
407  reference = FM17522_firmware_reference;
408  break;
409  case 0x90: // Version 0.0
410  reference = MFRC522_firmware_referenceV0_0;
411  break;
412  case 0x91: // Version 1.0
413  reference = MFRC522_firmware_referenceV1_0;
414  break;
415  case 0x92: // Version 2.0
416  reference = MFRC522_firmware_referenceV2_0;
417  break;
418  default: // Unknown version
419  return false; // abort test
420  }
421 
422  // Verify that the results match up to our expectations
423  for (i = 0; i < 64; i++) {
424  if (result[i] != pgm_read_byte(&(reference[i]))) {
425  return false;
426  }
427  }
428 
429  // Test passed; all is good.
430  return true;
431 } // End PCD_PerformSelfTest()

461 MFRC522::StatusCode MFRC522::PCD_CommunicateWithPICC( byte command,
462  byte waitIRq,
463  byte *sendData,
464  byte sendLen,
465  byte *backData,
466  byte *backLen,
467  byte *validBits,
468  byte rxAlign,
469  bool checkCRC
470  ) {
471  byte n, _validBits;
472  unsigned int i;
473 
474  // Prepare values for BitFramingReg
475  byte txLastBits = validBits ? *validBits : 0;
476  byte bitFraming = (rxAlign << 4) + txLastBits; // RxAlign = BitFramingReg[6..4]. TxLastBits = BitFramingReg[2..0]
477 
478  PCD_WriteRegister(CommandReg, PCD_Idle); // Stop any active command.
479  PCD_WriteRegister(ComIrqReg, 0x7F); // Clear all seven interrupt request bits
480  PCD_SetRegisterBitMask(FIFOLevelReg, 0x80); // FlushBuffer = 1, FIFO initialization
481  PCD_WriteRegister(FIFODataReg, sendLen, sendData); // Write sendData to the FIFO
482  PCD_WriteRegister(BitFramingReg, bitFraming); // Bit adjustments
483  PCD_WriteRegister(CommandReg, command); // Execute the command
484  if (command == PCD_Transceive) {
485  PCD_SetRegisterBitMask(BitFramingReg, 0x80); // StartSend=1, transmission of data starts
486  }
487 
488  // Wait for the command to complete.
489  // In PCD_Init() we set the TAuto flag in TModeReg. This means the timer automatically starts when the PCD stops transmitting.
490  // Each iteration of the do-while-loop takes 17.86�s.
491  i = 2000;
492  while (1) {
493  n = PCD_ReadRegister(ComIrqReg); // ComIrqReg[7..0] bits are: Set1 TxIRq RxIRq IdleIRq HiAlertIRq LoAlertIRq ErrIRq TimerIRq
494  if (n & waitIRq) { // One of the interrupts that signal success has been set.
495  break;
496  }
497  if (n & 0x01) { // Timer interrupt - nothing received in 25ms
498  return STATUS_TIMEOUT;
499  }
500  if (--i == 0) { // The emergency break. If all other conditions fail we will eventually terminate on this one after 35.7ms. Communication with the MFRC522 might be down.
501  return STATUS_TIMEOUT;
502  }
503  }
504 
505  // Stop now if any errors except collisions were detected.
506  byte errorRegValue = PCD_ReadRegister(ErrorReg); // ErrorReg[7..0] bits are: WrErr TempErr reserved BufferOvfl CollErr CRCErr ParityErr ProtocolErr
507  if (errorRegValue & 0x13) { // BufferOvfl ParityErr ProtocolErr
508  return STATUS_ERROR;
509  }
510 
511  // If the caller wants data back, get it from the MFRC522.
512  if (backData && backLen) {
513  n = PCD_ReadRegister(FIFOLevelReg); // Number of bytes in the FIFO
514  if (n > *backLen) {
515  return STATUS_NO_ROOM;
516  }
517  *backLen = n; // Number of bytes returned
518  PCD_ReadRegister(FIFODataReg, n, backData, rxAlign); // Get received data from FIFO
519  _validBits = PCD_ReadRegister(ControlReg) & 0x07; // RxLastBits[2:0] indicates the number of valid bits in the last received byte. If this value is 000b, the whole byte is valid.
520  if (validBits) {
521  *validBits = _validBits;
522  }
523  }
524 
525  // Tell about collisions
526  if (errorRegValue & 0x08) { // CollErr
527  return STATUS_COLLISION;
528  }
529 
530  // Perform CRC_A validation if requested.
531  if (backData && backLen && checkCRC) {
532  // In this case a MIFARE Classic NAK is not OK.
533  if (*backLen == 1 && _validBits == 4) {
534  return STATUS_MIFARE_NACK;
535  }
536  // We need at least the CRC_A value and all 8 bits of the last byte must be received.
537  if (*backLen < 2 || _validBits != 0) {
538  return STATUS_CRC_WRONG;
539  }
540  // Verify CRC_A - do our own calculation and store the control in controlBuffer.
541  byte controlBuffer[2];
542  MFRC522::StatusCode status = PCD_CalculateCRC(&backData[0], *backLen - 2, &controlBuffer[0]);
543  if (status != STATUS_OK) {
544  return status;
545  }
546  if ((backData[*backLen - 2] != controlBuffer[0]) || (backData[*backLen - 1] != controlBuffer[1])) {
547  return STATUS_CRC_WRONG;
548  }
549  }
550 
551  return STATUS_OK;
552 } // End PCD_CommunicateWithPICC()

649  // Byte 6: BCC Block Check Character - XOR of bytes 2-5
650  // Byte 7: CRC_A
651  // Byte 8: CRC_A
652  // The BCC and CRC_A are only transmitted if we know all the UID bits of the current Cascade Level.
653  //
654  // Description of bytes 2-5: (Section 6.5.4 of the ISO/IEC 14443-3 draft: UID contents and cascade levels)
655  // UID size Cascade level Byte2 Byte3 Byte4 Byte5
656  // ======== ============= ===== ===== ===== =====
657  // 4 bytes 1 uid0 uid1 uid2 uid3
658  // 7 bytes 1 CT uid0 uid1 uid2
659  // 2 uid3 uid4 uid5 uid6
660  // 10 bytes 1 CT uid0 uid1 uid2
661  // 2 CT uid3 uid4 uid5
662  // 3 uid6 uid7 uid8 uid9
663 
664  // Sanity checks
665  if (validBits > 80) {
666  return STATUS_INVALID;
667  }
668 
669  // Prepare MFRC522
670  PCD_ClearRegisterBitMask(CollReg, 0x80); // ValuesAfterColl=1 => Bits received after collision are cleared.
671 
672  // Repeat Cascade Level loop until we have a complete UID.
673  uidComplete = false;
674  while (!uidComplete) {
675  // Set the Cascade Level in the SEL byte, find out if we need to use the Cascade Tag in byte 2.
676  switch (cascadeLevel) {
677  case 1:
678  buffer[0] = PICC_CMD_SEL_CL1;
679  uidIndex = 0;
680  useCascadeTag = validBits && uid->size > 4; // When we know that the UID has more than 4 bytes
681  break;
682 
683  case 2:
684  buffer[0] = PICC_CMD_SEL_CL2;
685  uidIndex = 3;
686  useCascadeTag = validBits && uid->size > 7; // When we know that the UID has more than 7 bytes
687  break;
688 
689  case 3:
690  buffer[0] = PICC_CMD_SEL_CL3;
691  uidIndex = 6;
692  useCascadeTag = false; // Never used in CL3.
693  break;
694 
695  default:
696  return STATUS_INTERNAL_ERROR;
697  break;
698  }
699 
700  // How many UID bits are known in this Cascade Level?
701  currentLevelKnownBits = validBits - (8 * uidIndex);
702  if (currentLevelKnownBits < 0) {
703  currentLevelKnownBits = 0;
704  }
705  // Copy the known bits from uid->uidByte[] to buffer[]
706  index = 2; // destination index in buffer[]
707  if (useCascadeTag) {
708  buffer[index++] = PICC_CMD_CT;
709  }
710  byte bytesToCopy = currentLevelKnownBits / 8 + (currentLevelKnownBits % 8 ? 1 : 0); // The number of bytes needed to represent the known bits for this level.
711  if (bytesToCopy) {
712  byte maxBytes = useCascadeTag ? 3 : 4; // Max 4 bytes in each Cascade Level. Only 3 left if we use the Cascade Tag
713  if (bytesToCopy > maxBytes) {
714  bytesToCopy = maxBytes;
715  }
716  for (count = 0; count < bytesToCopy; count++) {
717  buffer[index++] = uid->uidByte[uidIndex + count];
718  }
719  }
720  // Now that the data has been copied we need to include the 8 bits in CT in currentLevelKnownBits
721  if (useCascadeTag) {
722  currentLevelKnownBits += 8;
723  }
724 
725  // Repeat anti collision loop until we can transmit all UID bits + BCC and receive a SAK - max 32 iterations.
726  selectDone = false;
727  while (!selectDone) {
728  // Find out how many bits and bytes to send and receive.
729  if (currentLevelKnownBits >= 32) { // All UID bits in this Cascade Level are known. This is a SELECT.
730  //Serial.print(F("SELECT: currentLevelKnownBits=")); Serial.println(currentLevelKnownBits, DEC);
731  buffer[1] = 0x70; // NVB - Number of Valid Bits: Seven whole bytes
732  // Calculate BCC - Block Check Character
733  buffer[6] = buffer[2] ^ buffer[3] ^ buffer[4] ^ buffer[5];
734  // Calculate CRC_A
735  result = PCD_CalculateCRC(buffer, 7, &buffer[7]);
736  if (result != STATUS_OK) {
737  return result;
738  }
739  txLastBits = 0; // 0 => All 8 bits are valid.
740  bufferUsed = 9;
741  // Store response in the last 3 bytes of buffer (BCC and CRC_A - not needed after tx)
742  responseBuffer = &buffer[6];
743  responseLength = 3;
744  }
745  else { // This is an ANTICOLLISION.
746  //Serial.print(F("ANTICOLLISION: currentLevelKnownBits=")); Serial.println(currentLevelKnownBits, DEC);
747  txLastBits = currentLevelKnownBits % 8;
748  count = currentLevelKnownBits / 8; // Number of whole bytes in the UID part.
749  index = 2 + count; // Number of whole bytes: SEL + NVB + UIDs
750  buffer[1] = (index << 4) + txLastBits; // NVB - Number of Valid Bits
751  bufferUsed = index + (txLastBits ? 1 : 0);
752  // Store response in the unused part of buffer
753  responseBuffer = &buffer[index];
754  responseLength = sizeof(buffer) - index;
755  }
756 
757  // Set bit adjustments
758  rxAlign = txLastBits; // Having a separate variable is overkill. But it makes the next line easier to read.
759  PCD_WriteRegister(BitFramingReg, (rxAlign << 4) + txLastBits); // RxAlign = BitFramingReg[6..4]. TxLastBits = BitFramingReg[2..0]
760 
761  // Transmit the buffer and receive the response.
762  result = PCD_TransceiveData(buffer, bufferUsed, responseBuffer, &responseLength, &txLastBits, rxAlign);
763  if (result == STATUS_COLLISION) { // More than one PICC in the field => collision.
764  byte valueOfCollReg = PCD_ReadRegister(CollReg); // CollReg[7..0] bits are: ValuesAfterColl reserved CollPosNotValid CollPos[4:0]
765  if (valueOfCollReg & 0x20) { // CollPosNotValid
766  return STATUS_COLLISION; // Without a valid collision position we cannot continue
767  }
768  byte collisionPos = valueOfCollReg & 0x1F; // Values 0-31, 0 means bit 32.
769  if (collisionPos == 0) {
770  collisionPos = 32;
771  }
772  if (collisionPos <= currentLevelKnownBits) { // No progress - should not happen
773  return STATUS_INTERNAL_ERROR;
774  }
775  // Choose the PICC with the bit set.
776  currentLevelKnownBits = collisionPos;
777  count = (currentLevelKnownBits - 1) % 8; // The bit to modify
778  index = 1 + (currentLevelKnownBits / 8) + (count ? 1 : 0); // First byte is index 0.
779  buffer[index] |= (1 << count);
780  }
781  else if (result != STATUS_OK) {
782  return result;
783  }
784  else { // STATUS_OK
785  if (currentLevelKnownBits >= 32) { // This was a SELECT.
786  selectDone = true; // No more anticollision
787  // We continue below outside the while.
788  }
789  else { // This was an ANTICOLLISION.
790  // We now have all 32 bits of the UID in this Cascade Level
791  currentLevelKnownBits = 32;
792  // Run loop again to do the SELECT.
793  }
794  }
795  } // End of while (!selectDone)
796 
797  // We do not check the CBB - it was constructed by us above.
798 
799  // Copy the found UID bytes from buffer[] to uid->uidByte[]
800  index = (buffer[2] == PICC_CMD_CT) ? 3 : 2; // source index in buffer[]
801  bytesToCopy = (buffer[2] == PICC_CMD_CT) ? 3 : 4;
802  for (count = 0; count < bytesToCopy; count++) {
803  uid->uidByte[uidIndex + count] = buffer[index++];
804  }
805 
806  // Check response SAK (Select Acknowledge)
807  if (responseLength != 3 || txLastBits != 0) { // SAK must be exactly 24 bits (1 byte + CRC_A).
808  return STATUS_ERROR;
809  }
810  // Verify CRC_A - do our own calculation and store the control in buffer[2..3] - those bytes are not needed anymore.
811  result = PCD_CalculateCRC(responseBuffer, 1, &buffer[2]);
812  if (result != STATUS_OK) {
813  return result;
814  }
815  if ((buffer[2] != responseBuffer[1]) || (buffer[3] != responseBuffer[2])) {
816  return STATUS_CRC_WRONG;
817  }
818  if (responseBuffer[0] & 0x04) { // Cascade bit set - UID not complete yes
819  cascadeLevel++;
820  }
821  else {
822  uidComplete = true;
823  uid->sak = responseBuffer[0];
824  }
825  } // End of while (!uidComplete)
826 
827  // Set correct uid->size
828  uid->size = 3 * cascadeLevel + 1;
829 
830  return STATUS_OK;
831 } // End PICC_Select()

1576  Serial.print(F(" "));

1581  c1 = buffer[7] >> 4;
1582  c2 = buffer[8] & 0xF;
1583  c3 = buffer[8] >> 4;
