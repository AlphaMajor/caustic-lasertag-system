110 void FragmentPlayer::playFragment(SoundSample* buffer)
111 {
112  m_buffer = buffer;
113  //printf("Enabling DMA\n");
114  DMA_DeInit(DMA2_Channel3);
115  m_DMA_InitStructure.DMA_MemoryBaseAddr = (uint32_t)buffer;
116  //printf("Fragment size: %u\n", m_size);
117  m_DMA_InitStructure.DMA_BufferSize = m_size;
118  DMA_Init(DMA2_Channel3, &m_DMA_InitStructure);
119  DMA_ITConfig(DMA2_Channel3, DMA_IT_TC, ENABLE);
120 
121  // Enable DMA2 Channel3
122  DMA_Cmd(DMA2_Channel3, ENABLE);
123 
124  DAC_Cmd(DAC_Channel_1, ENABLE);
125 
126  DAC_DMACmd(DAC_Channel_1, ENABLE);
127  //printf("Starting TIM6\n");
128  // TIM6 enable counter
129  TIM_Cmd(TIM6, ENABLE);
130 }
