90 void IRTransmitterTV::emitterCallback(bool wasOnState)
91 {
92  if (m_sendingHeader)
93  {
94  m_sendingHeader = false;
95  m_emitter->startImpulsePack(false, bitWaitPeriod);
96  return;
97  }
98 
99  if (wasOnState)
100  {
101  if (m_currentLength == m_length)
102  {
103  m_busy = false;
104  return;
105  }
106  m_emitter->startImpulsePack(false, bitWaitPeriod);
107  } else {
108  if (nextBit())
109  m_emitter->startImpulsePack(true, bitOnePeriod);
110  else
111  m_emitter->startImpulsePack(true, bitZeroPeriod);
112  }

168 void IRReceiverTV::saveBit(bool value)
169 {

171  return;
172 
173  m_haveSomeData = true;
174 
175  if (value)

177  else
178  *m_pCurrentByte &= ~(1 << m_currentBit);
179 
180  if (m_currentBit == 0) {
181  m_currentBit = 7;
182  m_pCurrentByte++;
183  } else

187 void IRReceiverTV::interruptHandler(bool state)
188 {
189  //printf("i\n");
190  Time time = systemClock->getTime();
191  uint32_t lastDtimeCandidate = m_dtime;
192  m_dtime = time - m_lastTime;
193  // Inverted input:
194  state = !state;
195  if (m_debug) {
196  printf("dt=%lu ", m_dtime );
197  if (state)
198  printf("1\n");
199  else
200  printf("0\n");
201  }
202 
203  if (m_dtime < 15) {
204  if (m_debug) printf ("short\n");
205  m_falseImpulse = true;
206  return;
207  }
208 
209 
210  if (m_falseImpulse) {
211  if (m_debug) printf ("short back\n");
212  m_falseImpulse = false;
213  return;
214  }
215  m_lastDtime = lastDtimeCandidate;
216  m_lastTime = time;
217 
218  switch(m_state) {
219  case RS_WAITING_HEADER:
220  // We should have 1
221  if (state == 0) {
222  resetReceiver();
223  return;
224  }
225 
226  if (m_debug) printf ("hb \n");
227  // Header beginned
228  m_state = RS_HEADER_BEGINNED;
229  break;
230 
231  case RS_HEADER_BEGINNED: {
232  // We should have 0
233  if (state != 0)
234  {
235  resetReceiver();
236  return;
237  }
238 
239  if (m_debug) printf ("he \n");
240  if (isCorrect(m_dtime, headerPeriodMin, headerPeriodMax)) {
241  if (m_debug) printf("ac \n");
242  m_state = RS_SPACE;
243 
244  } else {
245  resetReceiver();
246  return;
247  }
248  } break;
249 
250  case RS_SPACE: {
251  // We should have 1
252  if (state == 0)
253  {
254  resetReceiver();
255  return;
256  }
257 
258  if (m_debug) printf ("sp \n");
259  if (isCorrect(m_dtime, bitWaitPeriodMin, bitWaitPeriodMax+bitOnePeriodMax)) {
260  if (m_debug) printf("ac \n");
261  m_state = RS_BIT;
262  } else {
263  resetReceiver();
264  return;
265  }
266  } break;
267 
268  case RS_BIT: {
269  if (state != 0)
270  {
271  resetReceiver();
272  return;
273  }
274  if (m_debug) printf ("b \n");
275  if (isCorrect(m_dtime+m_lastDtime, bitOnePeriodMin+bitWaitPeriodMin, bitOnePeriodMax+bitWaitPeriodMax))
276  {
277  // We have bit "1"
278  if (m_debug) printf("ac 1 \n");
279  saveBit(true);
280  m_state = RS_SPACE;
281  } else if (isCorrect(m_dtime+m_lastDtime, bitZeroPeriodMin+bitWaitPeriodMin, bitZeroPeriodMax+bitWaitPeriodMax)){
282  // We have bit "0"
283  if (m_debug) printf("ac 0 \n");
284  saveBit(false);
285  m_state = RS_SPACE;
286  } else {
287  // Translation broken
288  resetReceiver();
289  return;
290  }
291  } break;
292  }
