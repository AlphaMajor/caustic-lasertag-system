124 void BluetoothBridge::receiveNetworkPackage(DeviceAddress sender, uint8_t* payload, uint16_t payloadLength)
125 {
126  debug << "Processing incoming network package";
127  m_bluetoothMsgCreator.clear();
128  m_bluetoothMsgCreator.setSender(std::move(sender));
129  m_bluetoothMsgCreator.addData(payloadLength, payload);
130  AnyBuffer* msgBuffer = new AnyBuffer(m_bluetoothMsgCreator.size(), m_bluetoothMsgCreator.data());
131  trace << "Bluetooth message to be sent: ";
132  printHex(msgBuffer->data, msgBuffer->size);
133  m_workerToBluetooth.add(
134  [this, msgBuffer] ()
135  {
136  sendBluetoothMessage(msgBuffer);
137  systemClock->wait_us(1000);
138  delete msgBuffer;
139  }
140  );
141 }

168 void BluetoothBridge::sendBluetoothMessage(AnyBuffer* buffer)
169 {
170  debug << "Sending bluetooth message";
171  // Transmitting to bluetooth module and waiting while transmit is done
172  m_bluetoothPort->transmit(buffer->data, buffer->size);
173  while (m_bluetoothPort->txBusy())
174  Kernel::yeld();
175 }

177 void BluetoothBridge::sendNetworkPackage(AnyBuffer* buffer)
178 {
179  debug << "Sending package to network";
180  // Dispatching bluetooth message
181  Message *bluetoothMessage = reinterpret_cast<Message *>(buffer->data);
182 
183  debug << "Bluetooth message for " << ADDRESS_TO_STREAM(bluetoothMessage->address);
184  // Sending message body as is
185  if (broadcast.isBroadcast(bluetoothMessage->address))
186  {
187  // We need special timings for broadcasts
188  NetworkLayer::instance().send(
189  bluetoothMessage->address,
190  bluetoothMessage->data,
191  bluetoothMessage->payloadLength(),
192  true,
193  nullptr,
194  bluetoothBridgePackageTimings.broadcast
195  );
196  } else {
197  // Not broadcast packages are with default timings
198  NetworkLayer::instance().send(bluetoothMessage->address, bluetoothMessage->data, bluetoothMessage->payloadLength(), true);
199  }
