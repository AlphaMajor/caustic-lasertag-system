88 void IOPin::enableExti(bool enable)
89 {
90  m_extiEnabled = enable;
91 
92  // Determining irq channel
93  IRQn_Type irq;
94 
95  switch(maskToPinNumber(m_pinMask))
96  {
97  default:
98  error << "Attempt to enable EXTI on invalid IOPin object";
99  case 0: irq = EXTI0_IRQn; break;
100  case 1: irq = EXTI1_IRQn; break;
101  case 2: irq = EXTI2_IRQn; break;
102  case 3: irq = EXTI3_IRQn; break;
103  case 4: irq = EXTI4_IRQn; break;
104  case 5: case 6: case 7: case 8: case 9:
105  irq = EXTI9_5_IRQn; break;
106  case 10: case 11: case 12: case 13: case 14: case 15:
107  irq = EXTI15_10_IRQn; break;
108  }
109 
110  if (m_extiEnabled)
111  {
112  GPIO_InitTypeDef GPIO_InitStructure;
113  zerify(GPIO_InitStructure);
114  GPIO_InitStructure.Pin = m_pinMask;
115  GPIO_InitStructure.Mode = GPIO_MODE_IT_RISING_FALLING;
116  GPIO_InitStructure.Speed = GPIO_SPEED_HIGH;
117  GPIO_InitStructure.Pull = GPIO_PULLUP;
118  HAL_GPIO_Init(m_gpio, &GPIO_InitStructure);
119 
120  HAL_NVIC_SetPriority(irq, 5, 0);
121  extisListeners[maskToPinNumber(m_pinMask)] = this;
122  HAL_NVIC_EnableIRQ(irq);
123  } else {
124  HAL_NVIC_DisableIRQ(irq);
125  switchToInput();
126  }
127 }

207 IIOPin* IOPinsPool::getIOPin(uint8_t portNumber, uint8_t pinNumber)
208 {
209  if (portNumber >= 8 || pinNumber >= 17)
210  return nullptr;
211  auto it = m_interfaces.find(std::pair<uint8_t, uint8_t>(portNumber, pinNumber));
212  if (it == m_interfaces.end())
213  {
214  IIOPin* newPin = new IOPin(portNumber, pinNumber);
215  m_interfaces[std::pair<uint8_t, uint8_t>(portNumber, pinNumber)] = newPin;
216  return newPin;
217  } else {
218  warning << "Double use of pin " << portNumber << ":" << pinNumber;
219  return it->second;
220  }
221 }
