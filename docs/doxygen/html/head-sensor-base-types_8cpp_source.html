30 void ConnectedWeaponsList::deserialize(void* source, OperationSize size)
31 {
32  trace << "Weapons list deserializing\n";
33  void* cursor = source;
34  uint8_t newSize = 0;
35  deserializeAndInc(cursor, newSize);
36  if (size != sizeof(newSize) + newSize*sizeof(DeviceAddress))
37  {
38  error << "Deserializing weapons list: invalid data size\n";
39  return;
40  }
41  m_weapons.clear();
42  for (uint8_t i=0; i<newSize; i++)
43  {
44  DeviceAddress addr;
45  deserializeAndInc(cursor, addr);
46  info << "Weapon restored: " << ADDRESS_TO_STREAM(addr);
47  insert(addr);
48  }
49 }
