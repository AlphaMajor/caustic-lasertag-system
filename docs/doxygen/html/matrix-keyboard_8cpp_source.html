46 uint8_t MatrixKeyboard::getKeyPressed()
47 {
48  if (systemClock->getTime() - m_lastKeypressTime < antiBounceDelay)
49  return keyNotPressed;
50  for (uint8_t i=0; i<m_rowCount; ++i)
51  {
52  m_row[i]->reset();
53  systemClock->wait_us(20);
54  for (uint8_t j=0; j<m_colCount; ++j)
55  {
56  if (m_col[j]->state() == false)
57  {
58  m_row[i]->set();
59  m_lastKeypressTime = systemClock->getTime();
60  return parseKey(i, j);
61  }
62  }
63  m_row[i]->set();
64  }
65  return keyNotPressed;
66 }

68 uint8_t MatrixKeyboard::parseKey(uint8_t row, uint8_t col)
69 {
70  if (row < 3 && col < 3)
71  {
72  return row*3 + (col+1);
73  }
74  if (col == 3)
75  {
76  return keyA + row;
77  }
78  if (row == 3)
79  {
80  switch (col)
81  {
82  case 0: return keyAsterisk;
83  case 1: return 0;
84  case 2: return keySharp;
85  default: return keyInvalid;
86  }
87  }
88  return keyInvalid;
