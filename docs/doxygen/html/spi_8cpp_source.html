88 void SPIManager::init(uint32_t prescaler, IIOPin* NSSPin, uint8_t SPIPhase)
89 {
90  m_stager.stage("init");
91  // May be there is no need to reconfigure hardware...
92  uint32_t hash = hashLy(prescaler, NSSPin, SPIPhase);
93  if (hash == m_configHash)
94  return;
95 
96  uint32_t baudratePrescaler;
97 
98  switch(prescaler)
99  {
100  case BaudRatePrescaler4: baudratePrescaler = SPI_BAUDRATEPRESCALER_4; break;
101  case BaudRatePrescaler8: baudratePrescaler = SPI_BAUDRATEPRESCALER_8; break;
102  case BaudRatePrescaler16: baudratePrescaler = SPI_BAUDRATEPRESCALER_16; break;
103  case BaudRatePrescaler32: baudratePrescaler = SPI_BAUDRATEPRESCALER_32; break;
104  case BaudRatePrescaler64: baudratePrescaler = SPI_BAUDRATEPRESCALER_64; break;
105  case BaudRatePrescaler128: baudratePrescaler = SPI_BAUDRATEPRESCALER_128; break;
106  case BaudRatePrescaler256: baudratePrescaler = SPI_BAUDRATEPRESCALER_256; break;
107  default:
108  error << "Invalid SPI prescaler";
109  return;
110  }
111 
112  m_hspi->Init.Mode = SPI_MODE_MASTER;
113  m_hspi->Init.Direction = SPI_DIRECTION_2LINES;
114  m_hspi->Init.DataSize = SPI_DATASIZE_8BIT;
115  m_hspi->Init.CLKPolarity = SPI_POLARITY_LOW;
116  m_hspi->Init.CLKPhase = SPIPhase == SPIPhase1edge ? SPI_PHASE_1EDGE : SPI_PHASE_2EDGE;
117  m_hspi->Init.NSS = SPI_NSS_SOFT;
118  m_hspi->Init.BaudRatePrescaler = baudratePrescaler;
119  m_hspi->Init.FirstBit = SPI_FIRSTBIT_MSB;
120  m_hspi->Init.TIMode = SPI_TIMODE_DISABLED;
121  m_hspi->Init.CRCCalculation = SPI_CRCCALCULATION_DISABLED;
122  m_hspi->Init.CRCPolynomial = 7;
123  HAL_SPI_Init(m_hspi);
124 
125 
126  m_NSSPin = NSSPin;
127  if (m_NSSPin)
128  {
129  m_NSSPin->switchToOutput();
130  m_NSSPin->set();
131  }
132 
133  m_configHash = hash;
134  m_stager.stage("init done");
135 }

137 bool SPIManager::Transmit(uint8_t *pData, uint16_t Size, uint32_t Timeout)
138 {
139  m_stager.stage("Transmit");
140 
141  m_operationDone = false;
142  taskENTER_CRITICAL();
143  HAL_StatusTypeDef res = HAL_SPI_Transmit_IT(m_hspi, pData, Size);
144  taskEXIT_CRITICAL();
145 
146  if (res != HAL_OK)
147  {
148  m_stager.stage("Transmit fail");
149  error << "SPI Transmit failed:" << res;
150  return false;
151  }
152  m_stager.stage("Transmit: waiting for it");
153  if (!waitForISR(Timeout))
154  {
155  error << "SPI Transmit timeout!";
156  return false;
157  }
158 
159  m_stager.stage("Transmit ok");
160  return true;
161 }

218 bool SPIManager::waitForISR(uint32_t timeout)
219 {
220  Time begin = systemClock->getTime();
221  while (!m_operationDone) {
222  if (systemClock->getTime() - begin > timeout)
223  {
224  // Not false to prevent case when context switching cause 'timeout',
225  // but not real timeout
226  return m_operationDone;
227  }
228  }
229  return true;
230 }
