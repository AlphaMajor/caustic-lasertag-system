100 
101  //debug << "ok";

103 
104  if (!m_contexts[channel].readHeader())

106  f_close(&m_contexts[channel].file);
107  error << "Invalid file header";
108  return false;
109  }
110 
111  //trace << "File header is ok";
112  m_contexts[channel].totalReaded = 0;
113 
114  // Adding first fragment from file to next buffer

155 bool WavPlayer::ChannelContext::readHeader()
156 {
157  FRESULT res;
158  UINT readed = 0;
159  res = f_read (&file, &header, sizeof(header), &readed);
160  if (res != FR_OK)
161  {
162  error << "Cannot read header from file";
163  return false;
164  }
165 
166  if (readed != sizeof(header))
167  {
168  error << "Incomplete wav file";
169  return false;
170  }
171 
172  //if (m_verbose) printInfo();
173 
174  // Validating m_header
175  if (strncmp(header.riff, "RIFF", 4) != 0)
176  {
177  error << "Invalid format: not RIFF";
178  return false;
179  }
180  if (header.audio_format != 1)
181  {
182  error << "Invalid audio format: " << header.audio_format << " - Only PCM audio supported";
183  return false;
184  }
185  if (header.num_channels != 1)
186  {
187  error << "Invalid channels count: " << header.num_channels << " - Only mono sound supported";
188  return false;
189  }
190  if (header.bits_per_sample != 16)
191  {
192  error << "Invalid bits per sample rate: " << header.bits_per_sample << " - Only 16 supported";
193  return false;
194  }
195  return true;
196 }

198 bool WavPlayer::loadFragment(SoundSample* buffer, uint8_t channel)
199 {
200  //trace << "loadig fragment...";
201  if (!m_contexts[channel].fileIsOpened)
202  {
203  //error << "Attempt to load fragment from closed file";
204  return false;
205  }
206 
207  FRESULT res;
208  UINT bytesReaded = 0;
209  int16_t *tmpPrt = reinterpret_cast<int16_t *>(m_tempBuffer);
210 
211  // For some unknown reason, f_read does not read more about 1k and crashes
213 
214  uint16_t sizeToRead = std::min(
215  audioBufferSize*sizeof(int16_t),
216  m_contexts[channel].header.subchunk2_size - m_contexts[channel].totalReaded
217  );
218 
219  res = f_read_huge(&m_contexts[channel].file, tmpPrt, sizeToRead, &bytesReaded);
220  if (res != FR_OK)
221  {
222  error << "Cannot read fragment from file: " << parseFRESULT(res);
223  f_close(&m_contexts[channel].file);
224  return false;
225  }
226 
227  m_contexts[channel].totalReaded += bytesReaded;
228 
229  if (bytesReaded < sizeToRead)
230  {
231  // We readed till file's end, so we can close it
232  //trace << "file's end: " << m_contexts[channel].totalReaded;
233  f_close(&m_contexts[channel].file);
234  m_contexts[channel].fileIsOpened = false;
235  }
236 
237 
238 
239  UINT samplesReaded = bytesReaded / sizeof(int16_t);
240 
241  for (unsigned int i=samplesReaded; i<audioBufferSize; i++)
242  {
243  tmpPrt[i] = 0;
244  }
245 
246 
247  // Now we have array of int16 samples so we need to add it to buffer
248  for (unsigned int i=0; i<audioBufferSize; i++)
249  {
250  buffer[i] += tmpPrt[i] >> 4;
251  }
252 
253  return true;
254 }
