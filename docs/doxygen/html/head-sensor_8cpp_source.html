286 void HeadSensor::catchShot(ShotMessage msg)
287 {
288  m_callbackStager.stage("shot");
289  info << "** Shot - team: " << msg.teamId << ", player: " << msg.playerId << ", damage: " << msg.damage;
290  Time currentTime = systemClock->getTime();
291  if (currentTime - m_shockDelayBegin < playerConfig.shockDelayImmortal)
292  {
293  info << "!! Shock time";
294  }
295  else if (playerState.isAlive()) {
296 
297  if (msg.playerId == playerConfig.plyerMT2Id)
298  {
299  debug << "self-shot";
300  msg.damage *= playerConfig.selfShotCoeff;
301  }
302  else if (msg.teamId == playerConfig.teamId)
303  {
304  debug << "friendly fire";
305  msg.damage *= playerConfig.frendlyFireCoeff;
306  }
307 
308  UintParameter healthBeforeDamage = playerState.healthCurrent;
309 
310  debug << "total damage: " << msg.damage;
311  playerState.damage(msg.damage);
312 
313  info << "health: " << playerState.healthCurrent << " armor: " << playerState.armorCurrent;
314 
315  // If still is alive
316  if (playerState.isAlive())
317  {
318  if (msg.damage != 0)
319  {
320  m_shockDelayBegin = systemClock->getTime();
321  weaponWoundAndShock();
322  m_statsCounter.registerHit(msg.playerId);
323  m_statsCounter.registerDamage(msg.playerId, msg.damage);
324  }
325  m_leds.blink(blinkPatterns.wound);
326  if (msg.playerId != playerConfig.plyerMT2Id)
327  {
328  notifyDamager(msg.playerId, msg.teamId, DamageNotification::injured);
329  }
330  } else {
331  //Player was killed
332  info << "xx Player died";
333  dieWeapons();
334  if (msg.playerId != playerConfig.plyerMT2Id)
335  {
336  notifyDamager(msg.playerId, msg.teamId, DamageNotification::killed);
337  }
338  m_statsCounter.registerKill(msg.playerId);
339  m_statsCounter.registerDamage(msg.playerId, healthBeforeDamage);
340  m_leds.blink(blinkPatterns.death);
342  //Scheduler::instance().addTask(std::bind(&StateSaver::saveState, &StateSaver::instance()), true, 0, 0, 1000000);
343  }
344  }
345 }
