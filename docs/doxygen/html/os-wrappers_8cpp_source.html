80 void TaskCycled::runTaskInCycle(void const* pTask)
81 {
82  TaskCycled *task = reinterpret_cast<TaskCycled*> (const_cast<void*>(pTask));
83  osDelay(task->m_firstRunDelay);
84  if (task->m_cyclesCount == 0)
85  {
86  for (;;)
87  {
88  ADD_BITS(task->m_state, executingNow);
89  task->m_task();
90  REMOVE_BITS(task->m_state, executingNow);
91  if (task->m_state & needStop)
92  {
93  task->stopThread();
94  return;
95  }
96  if (task->m_periodMin == 0)
97  taskYIELD();
98  else
99  osDelay(task->m_periodMin);
100  }
101  } else {
102  for (uint32_t i=0; i<task->m_cyclesCount;i++)
103  {
104  ADD_BITS(task->m_state, executingNow);
105  task->m_task();
106  REMOVE_BITS(task->m_state, executingNow);
107  if (task->m_state & needStop)
108  {
109  task->stopThread();
110  return;
111  }
112  if (task->m_periodMin == 0)
113  taskYIELD();
114  else
115  osDelay(task->m_periodMin);
116  }
117  task->stopThread();
118  }
119 }

216 bool TaskDeferredFromISR::run(STask&& task)
217 {
218  if (m_task != nullptr)
219  return false;
220 
221  m_task = task;
222 
223 
224  /* The actual processing is to be deferred to a task. Request the
225  vProcessInterface() callback function is executed, passing in the
226  number of the interface that needs processing. The interface to
227  service is passed in the second parameter. The first parameter is
228  not used in this case. */
229  BaseType_t xHigherPriorityTaskWoken = pdFALSE;
230  xTimerPendFunctionCallFromISR( &TaskDeferredFromISR::taskBody,
231  this,
232  sizeof(TaskDeferredFromISR),
233  &xHigherPriorityTaskWoken );
234 
235  /* If xHigherPriorityTaskWoken is now set to pdTRUE then a context
236  switch should be requested. The macro used is port specific and will
237  be either portYIELD_FROM_ISR() or portEND_SWITCHING_ISR() - refer to
238  the documentation page for the port being used. */
239  portYIELD_FROM_ISR( xHigherPriorityTaskWoken );
240  return true;
241 }

349 void TasksPool::main()
350 {
351  // Adding new tasks if exist
352  m_taskModifyMutex.lock();
353  for (auto it = m_tasksToAdd.begin(); it != m_tasksToAdd.end(); it = m_tasksToAdd.erase(it))
354  {
355  m_runningTasks.push_back(*it);
356  }
357  m_taskModifyMutex.unlock();
358 
359  // Running current tasks
360  for (auto it = m_runningTasks.begin(); it != m_runningTasks.end(); )
361  {
362  Time time = systemClock->getTime();
363  if (time >= it->timeToRun)
364  {
365  it->task();
366  it->repetitionsCount++;
367  it->timeToRun = time + it->period;
368  }
369 
370  time = systemClock->getTime();
371  if (
372  (it->repetitionsCountMax != 0 && it->repetitionsCount == it->repetitionsCountMax)
373  || (it->timeout != 0 && time > it->timeout)
374  )
375  {
376  it = m_runningTasks.erase(it);
377  } else
378  it++;
379  }
380 
381  // Removing deleted tasks
382  m_taskModifyMutex.lock();
383  for (auto it = m_idsToDelete.begin(); it != m_idsToDelete.end(); it++)
384  {
385  for (auto jt = m_runningTasks.begin(); jt != m_runningTasks.end(); )
386  {
387  if (jt->taskId == *it)
388  {
389  jt = m_runningTasks.erase(jt);
390  } else {
391  jt++;
392  }
393  }
394  }
395  m_idsToDelete.clear();
396  m_taskModifyMutex.unlock();
397 }
