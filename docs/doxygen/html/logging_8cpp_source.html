149 uint32_t Loggers::write(const char* buf, size_t nbyte)
150 {
151 
152  //ScopedLock lock(Loggers::loggersMutex);
153  if (isInitialized())
154  {
155  getUsart()->transmitSync((uint8_t*)buf, nbyte);
156  }
157  else
158  {
159  if (m_tmpBuffer == nullptr)
160  m_tmpBuffer = new char[tmpBufferSize];
161 
162  if (tmpBufferSize-m_tmpBufferCurrent > nbyte)
163  {
164  memcpy(&m_tmpBuffer[m_tmpBufferCurrent], buf, nbyte);
165  m_tmpBufferCurrent += nbyte;
166  }
167  }
168  return nbyte;
169 }
