84 void ButtonManager::extiCallback(bool state)
85 {
86  if (wasBounce() || !m_isEnabled) return;
87  uint32_t time = systemClock->getTime();
88  if (state == m_pressedState && m_isFirst == !m_pressedState && time - m_lastPressTime >= *m_repeatPeriod) {
89  m_extiDetected = true;

91  if (state == !m_pressedState) {

97 void ButtonManager::interrogate()
98 {
99  if (!m_inputInterrogator)
100  {
101  error << "Cannot interrogate button: m_inputInterrogator == nullptr";
102  return;
103  }
104 
105  if (!m_isEnabled) return;
106  if (wasBounce()) return;
107 
108  if (m_inputInterrogator->state() == m_pressedState || m_extiDetected == true)
109  {
110  // Button is pressed
111  uint32_t time = systemClock->getTime();
112  if (m_autoRepeat == true)
113  {
114  // Auto-repeating is enabled
115  if (time - m_lastPressTime >= *m_repeatPeriod)
116  {
117  // It was enough time since last call
118  m_lastPressTime = time;
119  m_callback(m_isFirst);
120  m_pressedAndNotDepressed = true;
121  m_isFirst = false;
122  }
123  // It was NOT enough time since last call
124  } else {
125  // Auto-repeating is disabled
126  if (time - m_lastPressTime >= *m_repeatPeriod && m_isFirst == true)
127  {
128  // It was enough time since last call
129  // Button was suddenly pressed
130  m_isFirst = false;
131  m_lastPressTime = time;
132  m_callback(true);
133  m_pressedAndNotDepressed = true;
134  }
135  }
136  }
137  if (m_inputInterrogator->state() == !m_pressedState) {
138  // Button depressed
139  m_isFirst = true;
140  if (m_pressedAndNotDepressed) {
141  m_pressedAndNotDepressed = false;
142  if (m_depressCallback)
143  m_depressCallback();
144  }
145  }
146  m_extiDetected = false;
147 }
