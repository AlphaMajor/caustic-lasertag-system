62 IAnyDevice* DeviceInitializer::initDevice(const char* filename)
63 {
64  IAnyDevice* resultDevice = nullptr;
65  if (isSdcardOk())
66  {
67  info << "Detecting device type using device.ini";
68  // We can read device type from config file
69  IniParcer* parcer = new IniParcer;
70  parcer->setCallback(
71  [&resultDevice] (const char* key, const char* value)
72  {
73  if (resultDevice != nullptr)
74  {
75  warning << "Config file should contain only ONE device_type record!";
76  return;
77  }
78  if (0 == strcmp(key, "device_type"))
79  {
80  if (0 == strcmp(value, "head_sensor"))
81  {
82  info << "Creating HEAD SENSOR device";
83  resultDevice = new HeadSensor;
84  }
85  else if (0 == strcmp(value, "rifle"))
86  {
87  info << "Creating RIFLE device";
88  resultDevice = new Rifle;
89  }
90  else if (0 == strcmp(value, "smart_point"))
91  {
92  info << "Creating SMART POINT device";
93  resultDevice = new SmartPoint;
94  }
95  else if (0 == strcmp(value, "bluetooth_bridge"))
96  {
97  info << "Creating BLUETOOTH BRIDGE device";
98  resultDevice = new BluetoothBridge;
99  }
100  else
101  resultDevice = nullptr;
102  }
103  }
104  );
105  Result res = parcer->parseFile(filename);
106  if (!res)
107  {
108  error << "Error while reading " << filename << ": " << res.errorText;
109  }
110  delete parcer;
111  } else {
112  info << "SD-card unavaliable";
113  info << "Creating BLUETOOTH BRIDGE device";
114  resultDevice = new BluetoothBridge;
115  }
116 
117  if (resultDevice == nullptr)
118  {
119  error << "Fatal error: unknown device type, aborting...\n";
120  for (;;);
121  }
122 
123 
124  info << "Reading device pinout...";
125  Pinout* pinout = new Pinout;
126 
127  bool pinoutReadingSucceeded = false;
128  if (isSdcardOk())
129  {
130  Result res = pinout->readIni("pinout.ini");
131  if (res)
132  {
133  pinoutReadingSucceeded = true;
134  } else {
135  error << "Error while reading pinout.ini: " << res.errorText;
136  }
137  }
138 
139  if (!pinoutReadingSucceeded)
140  {
141  error << "Cannot read device pinout, setting to default";
142  resultDevice->setDafaultPinout(*pinout);
143  }
144 
145  if (!resultDevice->checkPinout(*pinout))
146  {
147  error << "!! Fatal error: invalid pinout for selected device, cannot initialize device !!";
148  return nullptr;
149  }
150 
151  info << "Initializing device...";
152  resultDevice->init(*pinout);
153  delete pinout;
154  return resultDevice;
155 }
