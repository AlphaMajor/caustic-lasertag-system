159  ~Mutex() { if (handle) vSemaphoreDelete(handle); }
160 
161  inline bool lock(uint32_t timeout = portMAX_DELAY)
162  { return (xSemaphoreTake( handle, timeout ) == pdTRUE); }
163 
164  inline void unlock()
165  { xSemaphoreGive( handle ); }
166 
167  inline bool isLocked()
168  {
169  bool wasNotLocked = lock(0);
170  if (wasNotLocked) {
171  unlock();
172  return false;
173  }
174  return true;
175  }
176 private:
