42 void LEDFireEmitter::init(const Pinout& pinout)
43 {
45  // GPIO initialization - PWM output pins
46  RCC_APB2PeriphClockCmd(RCC_APB2Periph_GPIOA, ENABLE);
47  RCC_APB2PeriphClockCmd(RCC_APB2Periph_GPIOB, ENABLE);
48 
49  GPIO_InitTypeDef GPIO_InitStructure;
50  GPIO_StructInit(&GPIO_InitStructure);
51 
52  GPIO_InitStructure.GPIO_Pin = GPIO_Pin_6 | GPIO_Pin_7;
53  GPIO_InitStructure.GPIO_Mode = GPIO_Mode_AF_PP;
54  GPIO_InitStructure.GPIO_Speed = GPIO_Speed_50MHz;
55  GPIO_Init(GPIOA, &GPIO_InitStructure);
56 
57  GPIO_InitStructure.GPIO_Pin = GPIO_Pin_0 | GPIO_Pin_1;
58  GPIO_Init(GPIOB, &GPIO_InitStructure);
59 
61  // TIM3 initialization - 38/40/56kHz pulse through PWM
62  // Getting system clock frequency
63  RCC_APB1PeriphClockCmd(RCC_APB1Periph_TIM3, ENABLE);
64 
65  //Time initialization: setting frequency to default value (56KHz)
66  setCarrierFrequency(m_carrierFrequency);
67 
68  TIM_OC1PreloadConfig(TIM3, TIM_OCPreload_Enable);
69  TIM_OC2PreloadConfig(TIM3, TIM_OCPreload_Enable);
70  TIM_OC3PreloadConfig(TIM3, TIM_OCPreload_Enable);
71  TIM_OC4PreloadConfig(TIM3, TIM_OCPreload_Enable);
72 
73  modulationOff();
74 
76  // TIM7 initialization - "video impulse" former
77  RCC_ClocksTypeDef RCC_Clocks;
78  RCC_GetClocksFreq(&RCC_Clocks);
79  m_videoPrescaler = RCC_Clocks.HCLK_Frequency / 1000000;
80  RCC_APB1PeriphClockCmd(RCC_APB1Periph_TIM7, ENABLE);
81 
82  TIM_ClearITPendingBit(TIM7, TIM_IT_Update);
83 
84  NVIC_InitTypeDef NVIC_InitStructure;
85 
86  /* Configure the NVIC Preemption Priority Bits */
87  //NVIC_PriorityGroupConfig(NVIC_PriorityGroup_1);
88 
89  NVIC_InitStructure.NVIC_IRQChannel = TIM7_IRQn;
90  NVIC_InitStructure.NVIC_IRQChannelPreemptionPriority = 0;
91  NVIC_InitStructure.NVIC_IRQChannelSubPriority = 0;
92  NVIC_InitStructure.NVIC_IRQChannelCmd = ENABLE;
93  NVIC_Init(&NVIC_InitStructure);
94  //NVIC_SetPriority(TIM7_IRQn, 0);
95  //NVIC_EnableIRQ(TIM7_IRQn);
96 
97  //TIM_ITConfig(TIM7, TIM_DIER_UIE, ENABLE);
98 
99  // Creating power level channels
100  const char parameterNames[][25] = {
101  "IRPowerLevel1Channel",
102  "IRPowerLevel2Channel",
103  "IRPowerLevel3Channel",
104  "IRPowerLevel4Channel",
105  };
106  for (int i=0; i<4; i++)
107  {
108  auto result = pinout.getParameter(parameterNames[i]);
109  if (result) {
110  m_powerChannels[m_powerLevelsCount++] = StringParser<uint8_t>::parse(result.details.c_str());
111  }
112  }
113  setPower(100);
114 }

116 void LEDFireEmitter::startImpulsePack(bool isLedOn, unsigned int delayMs)
117 {
118  TIM_TimeBaseInitTypeDef TIM_TimeBaseInitStructure;
119  TIM_TimeBaseStructInit(&TIM_TimeBaseInitStructure);
120 
121  TIM_TimeBaseInitStructure.TIM_ClockDivision = TIM_CKD_DIV1;
122  TIM_TimeBaseInitStructure.TIM_CounterMode = TIM_CounterMode_Up;
123  TIM_TimeBaseInitStructure.TIM_Prescaler = m_videoPrescaler-1;
124  TIM_TimeBaseInitStructure.TIM_RepetitionCounter = 0;
125  TIM_TimeBaseInitStructure.TIM_Period = delayMs-1;
126  TIM_TimeBaseInit(TIM7, &TIM_TimeBaseInitStructure);
127 
128  TIM_ClearITPendingBit(TIM7, TIM_IT_Update);
129 
130  TIM_ITConfig(TIM7, TIM_DIER_UIE, ENABLE);
131 
132  m_isOn = isLedOn;
133  if (isLedOn)
134  modulationOn();
135  else
136  modulationOff();
137 
138  TIM_Cmd(TIM7, ENABLE);
139 }

172 void LEDFireEmitter::setChannel(unsigned int channel)
173 {
174  TIM_OCInitTypeDef TIM_OCInitStructure;
175  TIM_OCStructInit(&TIM_OCInitStructure);
176 
177  // Disabling all output channels
178  TIM_OCInitStructure.TIM_OCMode = TIM_OCMode_PWM1;
179  TIM_OCInitStructure.TIM_OutputState = TIM_OutputState_Disable;
180  TIM_OCInitStructure.TIM_Pulse = m_radioTimerPeriod / 2;
181  TIM_OCInitStructure.TIM_OCPolarity = TIM_OCPolarity_High;
182  TIM_OC1Init(TIM3, &TIM_OCInitStructure);
183  TIM_OC2Init(TIM3, &TIM_OCInitStructure);
184  TIM_OC3Init(TIM3, &TIM_OCInitStructure);
185  TIM_OC4Init(TIM3, &TIM_OCInitStructure);
186 
187  // Enabling one output channel
188  TIM_OCInitStructure.TIM_OutputState = TIM_OutputState_Enable;
189  m_PWMChannel = channel;
190  switch(m_PWMChannel)
191  {
192  default:
193  case 1: TIM_OC1Init(TIM3, &TIM_OCInitStructure); break;
194  case 2: TIM_OC2Init(TIM3, &TIM_OCInitStructure); break;
195  case 3: TIM_OC3Init(TIM3, &TIM_OCInitStructure); break;
196  case 4: TIM_OC4Init(TIM3, &TIM_OCInitStructure); break;
197  }
