60 void Stager::printStages() const
61 {
62  uint8_t current = m_pointer != 0 ? m_pointer-1 : ringBufferSize-1;
63 
64  debug << "Stages history: " << m_name;
65 
66  if (m_aliveCheckPeriod != 0)
67  {
68  if (m_stagesRingBuffer[current].t - systemClock->getTime() > m_aliveCheckPeriod)
69  debug << "Seems alive";
70  else
71  debug << "Seems dead!";
72  }
73 
74  for (uint8_t i=0; i<ringBufferSize; i++)
75  {
76  debug << i << ": [" << m_stagesRingBuffer[current].t << "] " << m_stagesRingBuffer[current].description;
77  if (current != 0)
78  current--;
79  else
80  current = ringBufferSize-1;
81  }
82 }
