87 void RC552Wrapper::readBlockLoop()
88 {
89  if (!cardReadyToOperate())
90  return;
91 
92  MFRC522::StatusCode status;
93 
94  // Authenticate using key A
95  debug << "Authenticating using key A...";
96  status = (MFRC522::StatusCode) m_mfrc.PCD_Authenticate(MFRC522::PICC_CMD_MF_AUTH_KEY_A, trailerBlock, &m_key, &(m_mfrc.uid));
97  if (status != MFRC522::STATUS_OK) {
98  error << "PCD_Authenticate() failed: " << m_mfrc.GetStatusCodeName(status);
99  return;
100  }
101 
102  debug << "Reading data from block " << blockAddr << " with size " << m_inputBufferSize;
103  status = (MFRC522::StatusCode) m_mfrc.MIFARE_Read(blockAddr, m_inputBuffer, &m_inputBufferSize);
104  if (status != MFRC522::STATUS_OK) {
105  error << "MIFARE_Read() failed: " << m_mfrc.GetStatusCodeName(status);
106  }
107 
108  // Halt PICC
109  m_mfrc.PICC_HaltA();
110  // Stop encryption on PCD
111  m_mfrc.PCD_StopCrypto1();
112 
113  debug << "Card reading done";
114 
115  if (m_readCallback)
116  {
117  m_readCallback(m_inputBuffer, m_inputBufferSize);
118  }
119 }

121 void RC552Wrapper::writeBlockLoop()
122 {
123  if (!cardReadyToOperate())
124  return;
125 
126  MFRC522::StatusCode status;
127 
128  // Authenticate using key B
129  status = (MFRC522::StatusCode) m_mfrc.PCD_Authenticate(MFRC522::PICC_CMD_MF_AUTH_KEY_B, trailerBlock, &m_key, &(m_mfrc.uid));
130  if (status != MFRC522::STATUS_OK) {
131  error << "PCD_Authenticate() failed: " << m_mfrc.GetStatusCodeName(status);
132  return;
133  }
134 
135  // Write data to the block
136  status = (MFRC522::StatusCode) m_mfrc.MIFARE_Write(blockAddr, m_dataToWrite, m_dataToWriteSize);
137  if (status != MFRC522::STATUS_OK) {
138  error << "MIFARE_Write() failed: " << m_mfrc.GetStatusCodeName(status);
139  }
140  //Serial.println();
141 
142  byte readedSize = std::max(18, (int) m_dataToWriteSize);
143  byte readBufferSize = readedSize;
144  // Verification
145  byte tmp[readedSize];
146 
147  // Read data from the block (again, should now be what we have written)
148  //Serial.print(F("Reading data from block ")); Serial.print(blockAddr);
149  //Serial.println(F(" ..."));
150  status = (MFRC522::StatusCode) m_mfrc.MIFARE_Read(blockAddr, tmp, &readedSize);
151  if (status != MFRC522::STATUS_OK) {
152  error << "MIFARE_Read() failed: " << m_mfrc.GetStatusCodeName(status);
153  }
154  if (readBufferSize < readedSize)
155  {
156  error << "MIFARE_Read() failed: readed " << readedSize << " instead of " << readBufferSize;
157  } else {
158  for (int i=0; i<m_dataToWriteSize; i++)
159  if (tmp[i] != m_dataToWrite[i])
160  {
161  error << "Data verification failed";
162  break;
163  }
164  }
165 
166  // Halt PICC
167  m_mfrc.PICC_HaltA();
168  // Stop encryption on PCD
169  m_mfrc.PCD_StopCrypto1();
170 
171  debug << "Card writing done";
172  if (m_writeDoneCallback)
173  {
174  m_writeDoneCallback(m_dataToWrite, m_dataToWriteSize);
175  m_writeDoneCallback = nullptr;
176  }
177 }

192  //Serial.println();

194  MFRC522::PICC_Type piccType = m_mfrc.PICC_GetType(m_mfrc.uid.sak);
195  debug << "PICC type: " << m_mfrc.PICC_GetTypeName(piccType);
196  //Serial.println(m_mfrc.PICC_GetTypeName(piccType));
197 
198  // Check for compatibility
199  if ( piccType != MFRC522::PICC_TYPE_MIFARE_MINI
200  && piccType != MFRC522::PICC_TYPE_MIFARE_1K
201  && piccType != MFRC522::PICC_TYPE_MIFARE_4K) {
202  error << "Reading only works with MIFARE Classic cards.";
203  return false;
204  }
205  return true;
