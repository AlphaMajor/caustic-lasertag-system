41 void Pinout::readConfigLine(const char* key, const char* value)
42 {
43  static const char portText[] = "_port";
44  static const char pinText[] = "_pin";
45  static const char invertedText[] = "_inverted";
46  static uint8_t portTextLen = strlen(portText);
47  static uint8_t pinTextLen = strlen(pinText);
48  uint8_t keyLen = strlen(key);
49  if (checkSuffix(key, portText)) // If we have port specification
50  {
51  std::string name = std::string(key).substr(0, keyLen-portTextLen);
52  auto it = m_pins.find(name);
54  uint8_t portNumber = atoi(value);
55  if (it != m_pins.end())
56  it->second.port = portNumber;
57  else
58  m_pins[name] = PinDescr(portNumber, 0);
59  } else if (checkSuffix(key, pinText)) // or pin specification
60  {
61  std::string name = std::string(key).substr(0, keyLen-pinTextLen);
62  auto it = m_pins.find(name);
64  uint8_t pinNumber = atoi(value);
65  if (it != m_pins.end())
66  it->second.pin = pinNumber;
67  else
68  m_pins[name] = PinDescr(0, pinNumber);
69  } else if (checkSuffix(key, invertedText)) // or 'inverted' specification
70  {
71  std::string name = std::string(key).substr(0, keyLen-pinTextLen);
72  bool isInverted = StringParser<bool>::parse(value);
73  auto it = m_pins.find(name);
74  if (it != m_pins.end())
75  it->second.inverted = isInverted;
76  else
77  m_pins[name] = PinDescr(0, 0, isInverted);
78  } else {
79  auto it = m_other.find(key);
80  if (it != m_other.end())
81  {
82  error << "Key duplication in pinout file: " << key;
83  }
84  m_other[key] = value;
85  }
86 }
