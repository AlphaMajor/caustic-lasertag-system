239  template<typename Type>
240  ResultType serializeCallRequest(
241  uint8_t* stream,
242  OperationCode functionCode,
243  uint16_t freeSpace,
244  uint16_t& actualSize,
245  const Type& parameter
246  )
247  {
249  actualSize = 0;
250  uint16_t packageSize = sizeof(OperationSize) + sizeof(OperationCode)+sizeof(parameter);
251  if (packageSize > freeSpace)
252  return DetailedResult<AddingResult>(NOT_ENOUGH_SPACE, "Not enough space in stream");
253 
254  functionCode = RCSPCodeManipulator::makeCallRequest(functionCode);
255  OperationSize size = sizeof(parameter);
256  memcpy(stream, &size, sizeof(OperationSize));
257  stream += sizeof(OperationSize);
258  memcpy(stream, &functionCode, sizeof(OperationCode));
259  stream += sizeof(OperationCode);
260  memcpy(stream, &parameter, sizeof(parameter));
261  actualSize = packageSize;
262  return DetailedResult<AddingResult>(OK);
263  }

278  template <typename T>
279  DetailedResult<T> getLocalObject(OperationCode code)
280  {
281  T result;
282  OperationCode objReq = RCSPCodeManipulator::makeSetObject(code);
283  auto it = m_accessorsByOpCode.find(code);
284  if (it == m_accessorsByOpCode.end())
285  return DetailedResult<T>(result, "Opcode not found");
286 
287  if (!it->second->isReadable())
288  return DetailedResult<T>(result, "Object not readable");
289 
290  if (it->second->getSize() != sizeof(result))
291  return DetailedResult<T>(result, "Object type size != required");
292 
293  it->second->serialize(&result);
294  return DetailedResult<T>(result);
295  }
