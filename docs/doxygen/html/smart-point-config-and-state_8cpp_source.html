63 void SmartPointState::ticTime()
64 {
65  if (currentTeam == teamGameIdNotATeam || gameState != gameStateInProcess)
66  return;
67 
68  Time currentTime = systemClock->getTime();
69  TimeInterval delta = currentTime - m_lastTimeTick;
70  m_lastTimeTick = currentTime;
71  switch(currentTeam)
72  {
73  case 0:
74  if (delta >= m_team1TimeLeftus)
75  {
76  m_team1TimeLeftus = 0;
77  win(currentTeam);
78  } else {
79  m_team1TimeLeftus -= delta;
80  }
81  break;
82 
83  case 1:
84  if (delta >= m_team2TimeLeftus)
85  {
86  m_team2TimeLeftus = 0;
87  win(currentTeam);
88  } else {
89  m_team2TimeLeftus -= delta;
90  }
91  break;
92 
93  case 2:
94  if (delta >= m_team3TimeLeftus)
95  {
96  m_team3TimeLeftus = 0;
97  win(currentTeam);
98  } else {
99  m_team3TimeLeftus -= delta;
100  }
101  break;
102 
103  case 3:
104  if (delta >= m_team4TimeLeftus)
105  {
106  m_team4TimeLeftus = 0;
107  win(currentTeam);
108  } else {
109  m_team4TimeLeftus -= delta;
110  }
111  break;
112  }
113  timeLeftUsToSec();
114 }
