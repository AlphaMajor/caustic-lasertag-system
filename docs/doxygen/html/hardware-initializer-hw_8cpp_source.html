50 void HardwareInitializer::SystemClock_Config()
51 {
52  DECL_CLEAN(RCC_OscInitTypeDef, RCC_OscInitStruct);
53  DECL_CLEAN(RCC_ClkInitTypeDef, RCC_ClkInitStruct);
54  DECL_CLEAN(RCC_PeriphCLKInitTypeDef, PeriphClkInit);
55  HAL_StatusTypeDef result = HAL_OK;
56 
57  HAL_SYSTICK_Config(HAL_RCC_GetHCLKFreq()/1000);
58 
59  HAL_SYSTICK_CLKSourceConfig(SYSTICK_CLKSOURCE_HCLK);
60 
61  // SysTick_IRQn interrupt configuration
62  HAL_NVIC_SetPriority(SysTick_IRQn, 0, 0);
63  char isLSEOk = 1;
64  RCC_OscInitStruct.OscillatorType = RCC_OSCILLATORTYPE_HSE|RCC_OSCILLATORTYPE_LSE;
65  RCC_OscInitStruct.HSEState = RCC_HSE_ON;
66  RCC_OscInitStruct.HSEPredivValue = RCC_HSE_PREDIV_DIV1;
67  RCC_OscInitStruct.LSEState = RCC_LSE_ON;
68  RCC_OscInitStruct.LSIState = RCC_LSI_OFF;
69  RCC_OscInitStruct.PLL.PLLState = RCC_PLL_ON;
70  RCC_OscInitStruct.PLL.PLLSource = RCC_PLLSOURCE_HSE;
71  RCC_OscInitStruct.PLL.PLLMUL = RCC_PLL_MUL9;
72  result = HAL_RCC_OscConfig(&RCC_OscInitStruct);
73  if (result != HAL_OK) {
74  isLSEOk = 0;
75  RCC_OscInitStruct.LSEState = RCC_LSE_OFF;
76  RCC_OscInitStruct.LSIState = RCC_LSI_ON;
77  HAL_RCC_OscConfig(&RCC_OscInitStruct);
78  }
79 
80  RCC_ClkInitStruct.ClockType = RCC_CLOCKTYPE_SYSCLK|RCC_CLOCKTYPE_PCLK1;
81  RCC_ClkInitStruct.SYSCLKSource = RCC_SYSCLKSOURCE_PLLCLK;
82  RCC_ClkInitStruct.AHBCLKDivider = RCC_SYSCLK_DIV1;
83  RCC_ClkInitStruct.APB1CLKDivider = RCC_HCLK_DIV2;
84  RCC_ClkInitStruct.APB2CLKDivider = RCC_HCLK_DIV1;
85  HAL_RCC_ClockConfig(&RCC_ClkInitStruct, FLASH_LATENCY_2);
86 
87 
88  PeriphClkInit.PeriphClockSelection = RCC_PERIPHCLK_RTC|RCC_PERIPHCLK_ADC;
89  if (isLSEOk)
90  PeriphClkInit.RTCClockSelection = RCC_RTCCLKSOURCE_LSE;
91  else
92  PeriphClkInit.RTCClockSelection = RCC_RTCCLKSOURCE_LSI;
93  PeriphClkInit.AdcClockSelection = RCC_ADCPCLK2_DIV8;
94  HAL_RCCEx_PeriphCLKConfig(&PeriphClkInit);
95 }
