38 void IRPresentationTransmitterMT2::sendMessage(const RCSPStream& stream)
39 {
40  RCSPAggregator::Operation op;
41  bool success = true;
42  RCSPAggregator::instance().extractNextOperation(stream.getStream(), op, stream.getSize(), success);
43  switch(op.code)
44  {
45  case ConfigCodes::HeadSensor::Functions::catchShot: {
46  const ShotMessage* shot = reinterpret_cast<const ShotMessage*>(op.argument);
47  m_data[0] = shot->playerId & 0b01111111;
48  m_data[1] = (shot->teamId << 6) | ( (encodeDamage(shot->damage) & 0b00001111) << 2);
49  m_physicalTransmitter->send(m_data, MT2Extended::shotLength);
50  } break;
51  case ConfigCodes::HeadSensor::Functions::playerKill: {
52  sendCommand(MT2Extended::Commands::adminKill);
53  } break;
54  case ConfigCodes::HeadSensor::Functions::playerReset: {
55  sendCommand(MT2Extended::Commands::startGame);
56  } break;
57  case ConfigCodes::AnyDevice::Functions::resetToDefaults: {
58  sendCommand(MT2Extended::Commands::restoreDefaults);
59  } break;
60  case ConfigCodes::HeadSensor::Functions::playerRespawn: {
61  sendCommand(MT2Extended::Commands::respawn);
62  } break;
63  default:
64  sendGenericRCSPMessageDefault(stream);
65  }

82 uint8_t IRPresentationTransmitterMT2::encodeDamage(uint8_t damage)
83 {
84  if (damage == 1)
85  return 0;
86 
87  if (damage == 2)
88  return 1;
89 
90  if (damage >= 3 && damage < 5)
91  return 2;
92 
93  if (damage == 5)
94  return 3;
95 
96  if (damage >= 6 && damage < 9)
97  return 4;
98 
99  if (damage >= 9 && damage < 14)
100  return 5;
101 
102  if (damage >= 14 && damage < 16)
103  return 6;
104 
105  if (damage >= 16 && damage < 19)
106  return 7;
107 
108  if (damage >= 19 && damage < 23)
109  return 8;
110 
111  if (damage >= 23 && damage < 28)
112  return 9;
113 
114  if (damage >= 28 && damage < 33)
115  return 10;
116 
117  if (damage >= 33 && damage < 38)
118  return 11;
119 
120  if (damage >= 38 && damage < 45)
121  return 12;
122 
123  if (damage >= 45 && damage < 58)
124  return 13;
125 
126  if (damage >= 58 && damage < 88)
127  return 14;
128 
129  if (damage >= 88)
130  return 15;
131 
132  // To avoid warning
133  return 0;

171 void IRPresentationReceiverMT2::receiverCallback(const uint8_t* data, uint16_t size)
172 {
173  debug << "IR receiver has data: ";
174  printHex(data, size);
175  // If shot
176  if ((data[0] & ~MT2Extended::Byte1::shotMask) == 0)
177  {
178  // For compatibility with protocol extensions no limitations to package size are provided
179  parseShot(data, size);
180  return;
181  }
182  // Parsing a "command"
183  if (size == MT2Extended::commandLength
184  && data[0] == MT2Extended::Byte1::command
185  && data[2] == MT2Extended::Byte3::commandEnd)
186  {
187  parseMT2Command(data, size);
188  return;
189  }
190 
191  // Team change
192  if (size == MT2Extended::messageLength
193  && data[0] == MT2Extended::Byte1::setTeam)
194  {
195  parseSetTeam(data, size);
196  return;
197  }
198 
199  // Health count changing
200  if (size == MT2Extended::messageLength
201  && data[0] == MT2Extended::Byte1::addHealth)
202  {
203  parseAddHealth(data, size);
204  return;
205  }
206 
207  if (size > 8 + RCSPAggregator::minimalStreamSize
208  && size % 8 == 0
209  && data[0] == MT2Extended::Byte1::RCSPMessage)
210 
211  {
212  parseRCSP(data, size);
213  return;
214  }
215

218 void IRPresentationReceiverMT2::parseShot(const uint8_t* data, uint16_t size)
219 {
220  UNUSED_ARG(size);
221  if (size < MT2Extended::shotLength)
222  {
223  warning << "Shot package with len < " << size << " ignored";
224  return;
225  }
226  ShotMessage msg;
227  msg.playerId = data[0] & ~(1 << 7);
228  msg.teamId = data[1] >> 6;
229  unsigned int damageCode = (data[1] & 0b00111100) >> 2;
230  msg.damage = decodeDamage(damageCode) * *m_damageCoefficient;
231  if (!m_group->m_hasWaitingShot || msg.damage > m_group->m_maxDamage)
232  {
233  m_group->m_shotMessage = msg;
234  m_group->m_hasWaitingShot = true;
235  updateOperationCallback(
236  [this]() {
237  RCSPAggregator::instance().doOperation(ConfigCodes::HeadSensor::Functions::catchShot, m_group->m_shotMessage);
238  }
239  );
240  }
241 }

314 void IRPresentationReceiverMT2::parseRCSP(const uint8_t* data, uint16_t size)
315 {
316  const uint8_t* stream = &data[1];
317  uint16_t streamSize = size-1;
318  if (!RCSPAggregator::instance().isStreamConsistent(stream, streamSize))
319  {
320  warning << "Inconsistent RCSP stream on IR channel";
321  return;
322  }
323  debug << "General RCSP message by IR receiver";
324  if (size > PresentationReceiversGroupMT2::argumentBufferSize)
325  {
326  error << "RCSP IR stream is too large to be copied to temporary buffer";
327  return;
328  }
329  memcpy(m_group->m_bufferForArgument, stream, streamSize);
330  updateOperationCallback([this, streamSize]() mutable {
331  RCSPAggregator::instance().dispatchStream(m_group->m_bufferForArgument, streamSize);
332  });
333 }
