187 template<>
188 class StringParser<bool>
189 {
190 public:
191  static bool isSupported() { return true; }
192 
193  static bool parse(const char* str)
194  {
195  unsigned int cursor = 0;
196  while (isSpace(str[cursor])) cursor++;
197  const char* begin = &str[cursor];
198 
199  while (!isSpace(str[cursor]) && (str[cursor] != '\0')) cursor++;
200  unsigned int len = &str[cursor] - begin;
201 
202  if (len == 1)
203  {
204  if (begin[0] == '1')
205  return true;
206  }
207 
208  if (len == 2)
209  {
210  if (lower(begin[0]) == 'o'
211  && lower(begin[1]) == 'n')
212  return true;
213  }
214 
215  if (len == 4)
216  {
217  if (lower(begin[0]) == 't'
218  && lower(begin[1]) == 'r'
219  && lower(begin[2]) == 'u'
220  && lower(begin[3]) == 'e')
221  return true;
222  }
223 
224  return false;
225  }
226 };
