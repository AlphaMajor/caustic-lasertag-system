76 void BaseStatsCounter::saveState()
77 {
78  FRESULT res = FR_OK;
79  info << "<Save stats> Saving stats";
80  res = f_open(&m_file, filename, FA_CREATE_ALWAYS | FA_WRITE);
81  if (res != FR_OK)
82  {
83  error << "<Save stats> Cannot open file to write base stats: " << res;
84  return;
85  }
86 
87  for (auto it = m_results.begin(); it != m_results.end(); it++)
88  {
89  UINT written = 0;
90  res = f_write(&m_file, & (it->second), sizeof(PvPDamageResults), &written);
91  if (res != FR_OK)
92  {
93  error << "<Save stats> Cannot write base stats to file: " << res;
94  break;
95  }
96  if (written != sizeof(PvPDamageResults))
97  {
98  error << "<Save stats> Wrong size written to file: " << written << " instead of " << sizeof(PvPDamageResults);
99  break;
100  }
101  }
102  f_close(&m_file);
103 }

159 void BaseStatsCounter::sendNextPackage()
160 {
161  debug << "BaseStatsCounter::sendNextPackage()";
162  ScopedLock<CritialSection> lock(m_iteratorCheck);
163  if (m_iteratorCorrupted)
164  {
165  prepareTransmission();
166  return;
167  }
168  // Iterator is fine here
169  if (m_sendingIterator == m_results.end())
170  {
171  // All data transmitted
172  m_sendingState = S_NOTHING;
173  return;
174  }
175 
176  PvPDamageResults target = m_sendingIterator->second;
177  m_sendingIterator++;
178  // Now we have valid object and we can send it
179  lock.unlock();
180  debug << "Sending from " << target.enemyId;
181  m_sendingState = S_WAITING_FOR_TRANSMISSIO_RESULT;
182  RCSPStream::remoteCall(
183  m_statsReceiver,
184  ConfigCodes::Base::Functions::getPvPResults,
185  target,
186  true,
187  [this](PackageId, bool isSuccess) {
188  if (isSuccess)
189  onTransmissionSucceeded();
190  else
191  onTransmissionBroken();
192  }
193  );
194 
195 }
