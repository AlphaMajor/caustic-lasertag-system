52 uint32_t RCSPAggregator::dispatchStream(uint8_t* stream, uint32_t size, RCSPMultiStream* answerStream)
53 {
54  uint8_t* position = stream;
55  uint32_t unsupported = 0;
56  while ((position - stream) <= size - (sizeof(OperationSize)+sizeof(OperationCode)) )
57  {
58  OperationSize *pOperationSize = reinterpret_cast<OperationSize*> (position);
59  position += sizeof(OperationSize);
60  OperationCode *pOperationCode = reinterpret_cast<OperationCode*> (position);
61  position += sizeof(OperationCode);
62  if (!dispatchOperation(pOperationSize, pOperationCode, position, answerStream))
63  unsupported++;
64  position += *pOperationSize;
65  }
66  return unsupported;
67 }

93 bool RCSPAggregator::dispatchOperation(OperationSize* size, OperationCode* code, uint8_t* arg, RCSPMultiStream* answerStream)
94 {
95  if (RCSPCodeManipulator::isObjectRequest(*code))
96  {
97  if (answerStream)
98  {
99  // Adding parameter to answer stream
100  OperationCode parameterCode = RCSPCodeManipulator::makeSetObject(*code);
101  //printf("Parameter request: %u\n", parameterCode);
102  auto it = m_accessorsByOpCode.find(parameterCode);
103  if (it != m_accessorsByOpCode.end())
104  {
105  answerStream->addValue(parameterCode);
106  return true;
107  } else {
108  printWarningUnknownCode(*code);
109  return false;
110  }
111  } else {
112  debug << "No answer stream, skipping request";
113  return false;
114  }
115  } else {
116  auto it = m_accessorsByOpCode.find(*code);
117  if (it != m_accessorsByOpCode.end())
118  {
119  trace << "Dispatched opcode: " << *code ;
120  if (*size == 0)
121  it->second->deserialize(nullptr, 0);
122  else
123  it->second->deserialize(arg, *size);
124  return true;
125  }
126  else
127  {
128  printWarningUnknownCode(*code);
129  return false;
130  }
131  }
132 }

142 bool RCSPAggregator::isStreamConsistent(const uint8_t* stream, uint32_t size)
143 {
144  //uint8_t* position = stream;
145  for (const uint8_t* position = stream; position - stream <= size; position++)
146  {
147  OperationSize operationSize = *position;
148  unsigned int thisOperationSize = sizeof(operationSize)+sizeof(OperationCode)+operationSize;
149  position += thisOperationSize;
151  if (position - stream == size)
152  return true;
153  else if (position - stream > size)
154  return false;

178 DetailedResult<RCSPAggregator::AddingResult> RCSPAggregator::serializeObject(
179  uint8_t* stream,
180  OperationCode code,
181  uint16_t freeSpace,
182  uint16_t& actualSize,
183  const uint8_t* pCustomValue)
184 {
185  actualSize = 0;
186  auto it = m_accessorsByOpCode.find(code);
187  if (it == m_accessorsByOpCode.end())
188  {
189  return DetailedResult<AddingResult>(INVALID_OPCODE, "Opcode not found");
190  }
191  code = RCSPCodeManipulator::makeSetObject(code);
192  OperationSize size = 0;
193  if (!it->second->isReadable())
194  return DetailedResult<AddingResult>(NOT_READABLE, "Object not readable");
195 
196  size = it->second->getSize();
197  uint16_t packageSize = size + sizeof(OperationSize) + sizeof(OperationCode);
198  if (freeSpace < packageSize)
199  {
200  return DetailedResult<AddingResult>(NOT_ENOUGH_SPACE, "Not enough space in stream");
201  }
202  memcpy(stream, &size, sizeof(OperationSize));
203  stream += sizeof(OperationSize);
204  memcpy(stream, &code, sizeof(OperationCode));
205  stream += sizeof(OperationCode);
206  if (pCustomValue == nullptr)
207  {
208  it->second->serialize(stream);
209  } else {
210  memcpy(stream, pCustomValue, size);
211  }
212 
213  actualSize = sizeof(OperationSize) + sizeof(OperationCode) + size;
214  return DetailedResult<AddingResult>(OK);
215 }
