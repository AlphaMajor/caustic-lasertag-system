110 UIStateBase* UIInitScreen::doIteration()
111 {
112  if (!m_screenUpdated)
113  {
114  m_ui.lcd.configureSPI();
115  m_ui.lcd.clearBuffer();
116  m_ui.lcd.setFont(LCD5110Controller::fontStandardAscii5x7);
117  m_ui.lcd.stringXY(10, 0, "Caustic LTS");
118  m_ui.lcd.stringXY(0, 1, "Select mode:");
119  m_ui.lcd.stringXY(0, 2, "A: Start");
120  m_ui.lcd.stringXY(0, 3, "B: Program");
121  char buffer[20];
122  sprintf(buffer, "1/2: Cont %d", (int) m_ui.lcd.lcdContrast);
123  m_ui.lcd.stringXY(0, 5, buffer);
124  m_ui.lcd.write();
125  m_screenUpdated = true;
126  }
127 
128 
129  uint8_t key = m_ui.keyboard.getKeyPressed();
130  if (key == 'A')
131  {
132  m_ui.smartPointState.beginGame();
133  m_ui.timeLeftScreen.prepare();
134  return &m_ui.timeLeftScreen;
135  }
136 
137  if (key == 1)
138  {
139  if (m_ui.lcd.lcdContrast > 1)
140  {
141  m_ui.lcd.lcdContrast--;
142  m_ui.lcd.updateContrast();
143  debug << "Contrast dec";
144  m_screenUpdated = false;
145  }
146  }
147 
148  if (key == 2)
149  {
150  if (m_ui.lcd.lcdContrast < 127)
151  {
152  m_ui.lcd.lcdContrast++;
153  m_ui.lcd.updateContrast();
154  debug << "Contrast inc";
155  m_screenUpdated = false;
156  }
157  }
158 
159 
160  return this;
161 }

172 {

188 
189 
190  m_ui.lcd.stringXY(0, 2, "Yellow:");
191  formatTime(buffer, m_ui.smartPointState.team3TimeLeft / 60, m_ui.smartPointState.team3TimeLeft % 60);
192  m_ui.lcd.stringXY(40, 2, buffer);
193 
194 
195  m_ui.lcd.stringXY(0, 3, "Green:");
196  formatTime(buffer, m_ui.smartPointState.team4TimeLeft / 60, m_ui.smartPointState.team4TimeLeft % 60);
197  m_ui.lcd.stringXY(40, 3, buffer);
198 
199  if (m_winRegisered)
200  {
201  switch (m_ui.smartPointState.currentTeam)
202  {
203  case 0: m_ui.lcd.stringXY(0, 5, "Red wins!"); break;
204  case 1: m_ui.lcd.stringXY(0, 5, "Blue wins!"); break;
205  case 2: m_ui.lcd.stringXY(0, 5, "Yellow wins!"); break;
206  case 3: m_ui.lcd.stringXY(0, 5, "Green wins!"); break;
207  default: break;
208  }
209  }
210  m_ui.lcd.write();
211 
212  //m_screenUpdated = true;
213  }
214 
215  // If leader team changed
216  if (m_lastLeader != m_ui.smartPointState.currentTeam)
217  {
218  m_lastLeader = m_ui.smartPointState.currentTeam;
219 
220  m_ui.leds.blink(SmartPointUI::patterns.teamHasPoint);
221  }
222 
223  if (!m_winRegisered && m_ui.smartPointState.gameState == SmartPointState::gameStateEnd)
224  {
225  m_ui.leds.blink(SmartPointUI::patterns.teamWin);
226  m_winRegisered = true;
227  }
228 
229  uint8_t key = m_ui.keyboard.getKeyPressed();
230  if (key != MatrixKeyboard::keyNotPressed)
231  {
232  if (key == '#')
233  {
234  m_ui.smartPointState.stopGame();
235  m_ui.leds.blink(SmartPointUI::patterns.hello, RGBLeds::red | RGBLeds::green | RGBLeds::blue);
236  m_ui.initScreen.prepare();
237  return &m_ui.initScreen;
238  }
239  }
240 
241  return this;
242 }
