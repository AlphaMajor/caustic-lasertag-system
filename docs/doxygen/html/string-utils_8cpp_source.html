38 void formatTime(char* buffer, uint16_t min, uint16_t sec)
39 {
40  unsigned int i=0;
41  if (min == 0)
42  {
43  buffer[i++] = '0';
44  } else {
45  for (; min != 0; min /= 10)
46  {
47  buffer[i++] = min % 10 + '0';
48  }
49 
50  for (int j=0; j<i/2; j++)
51  {
52  std::swap(buffer[j], buffer[i-1-j]);
53  }
54  }
55 
56  buffer[i++] = ':';
57  buffer[i++] = sec / 10 + '0';
58  buffer[i++] = sec % 10 + '0';
59  buffer[i] = '\0';

67 void doubleToString(char* buffer, double value, int bufferSize, int fracDigits)
68 {
69  unsigned int digits = value * 100.0;
70  int pointPosition = 0;
71 
72  for (int p = bufferSize-1; p>=0; p--)
73  {
74  if (digits == 0 && pointPosition-p > 1)
75  buffer[p] = ' ';
76  else
77  {
78  buffer[p] = digits % 10 + '0';
79  digits /= 10;
80  }
81 
82  if (p == bufferSize-1 - 1)
83  {
84  buffer[--p] = '.';
85  pointPosition = p;
86  }
87  }

151 Result IniParcer::parseFile(const char* filename)
152 {
153  if (!m_acceptKeyValueCallback)
154  return Result("Callback not set");
155 
156  FIL* m_file = new FIL;
157 
158  FRESULT fres = f_open(m_file, filename, FA_OPEN_EXISTING | FA_READ);
159  if (fres != FR_OK)
160  {
161  return Result("Cannot open file\n");
162  }
163 
164  Result result;
165  ParcerState state = PS_WAITING_KEY;
166  unsigned int keyCursor = 0, valueCursor = 0;
167  unsigned int line = 1;
168  bool error = false;
169 
170  while (!f_eof(m_file) && !error)
171  {
172  UINT readed = 0;
173  unsigned int cursor = 0;
174  fres = f_read(m_file, buffer, blockSize, &readed);
175  if (fres != FR_OK) {
176  return Result(parseFRESULT(fres));
177  }
178  // Skipping spaces
179  while (cursor < readed && !error)
180  {
181  switch (state)
182  {
183  case PS_COMMENT:
184  while (cursor < readed && buffer[cursor] != '\n')
185  cursor++;
186 
187  if (cursor != readed) {
188  state = PS_WAITING_KEY;
189  cursor++;
190  line++;
191  }
192  break;
193 
194 
195  case PS_WAITING_KEY:
196  for (; cursor < readed; cursor++)
197  {
198  if (buffer[cursor] == '#')
199  {
200  state = PS_COMMENT;
201  cursor++;
202  break;
203  }
204 
205  if (buffer[cursor] == '\n')
206  {
207  line++;
208  continue;
209  }
210 
211  if (isCharacter(buffer[cursor]))
212  {
213  state = PS_READING_KEY;
214  break;
215  }
216  if (!isSpace(buffer[cursor]))
217  {
218  sprintf(m_errorMessage, "Expected key at line %d\n", line);
219  result.fail(m_errorMessage);
220  error = true;
221  break;
222  }
223  }
224  break;
225 
226  case PS_READING_KEY:
227  for (; cursor < readed; cursor++)
228  {
229  if (buffer[cursor] == '=' || isSpace(buffer[cursor]))
230  {
231  state = PS_WAITING_EQ;
232  key[keyCursor] = '\0';
233  break;
234  }
235 
236  if (buffer[cursor] == '\n')
237  {
238  sprintf(m_errorMessage, "Expected \'= value\' at line %d\n", line);
239  result.fail(m_errorMessage);
240  error = true;
241  break;
242  }
243 
244  if (!isCharacter(buffer[cursor]) && !isNumber(buffer[cursor]))
245  {
246  sprintf(m_errorMessage, "Unexpected character in key name at line %d\n", line);
247  result.fail(m_errorMessage);
248  error = true;
249  break;
250  }
251  key[keyCursor++] = buffer[cursor];
252  if (keyCursor == keyValMaxLength-2)
253  {
254  sprintf(m_errorMessage, "Key is too long at line %d\n", line);
255  result.fail(m_errorMessage);
256  error = true;
257  break;
258  }
259  }
260  break;
261 
262 
263  case PS_WAITING_EQ:
264  for (; cursor < readed; cursor++)
265  {
266  if (buffer[cursor] == '=')
267  {
268  state = PS_WAITING_VALUE;
269  cursor++;
270  break;
271  }
272 
273  if (!isSpace(buffer[cursor]))
274  {
275  sprintf(m_errorMessage, "Expexted \'=\' symbol at line %d\n", line);
276  result.fail(m_errorMessage);
277  error = true;
278  break;
279  }
280  }
281  break;
282 
283  case PS_WAITING_VALUE:
284  for (; cursor < readed; cursor++)
285  {
286  if (isCharacter(buffer[cursor]) || isNumber(buffer[cursor]))
287  {
288  state = PS_READING_VALUE;
289  break;
290  }
291  if (!isSpace(buffer[cursor]))
292  {
293  sprintf(m_errorMessage, "Expexted value at line %d\n", line);
294  result.fail(m_errorMessage);
295  error = true;
296  break;
297  }
298  }
299  break;
300 
301  case PS_READING_VALUE:
302  for (; cursor < readed; cursor++)
303  {
304  if (buffer[cursor] == '\n')
305  {
306  state = PS_WAITING_KEY;
307  value[valueCursor] = '\0';
308  cursor++;
309  line++;
310  m_acceptKeyValueCallback(key, value);
311  valueCursor = 0;
312  keyCursor = 0;
313  break;
314  }
315 
316  value[valueCursor++] = buffer[cursor];
317  if (valueCursor == keyValMaxLength-2)
318  {
319  sprintf(m_errorMessage, "Value is too long at line %d\n", line);
320  result.fail(m_errorMessage);
321  error = true;
322  break;
323  }
324  }
325  break;
326  }
327  }
328  }
329  f_close(m_file);
330  delete m_file;
331  return result;
332 }
