117 void UARTManager::rxDoneISR(uint8_t* buffer, uint16_t size)
118 {
119  UNUSED_ARG(buffer);
120  UNUSED_ARG(size);
121  rxCount++;
122  if (m_stopCharEnabled && rxBuffer[rxCount-1] == m_stopChar)
123  {
124  m_rxCallback(rxBuffer, rxCount);
125  rxCount = 0;
126  } else if (rxCount == rxBufferMaxSize || rxCount == m_blockSize)
127  {
128  m_rxCallback(rxBuffer, rxCount);
129  rxCount = 0;
130  }
131 
132  HAL_UART_Receive_IT(m_huart, &(rxBuffer[rxCount]), 1);
133 }
