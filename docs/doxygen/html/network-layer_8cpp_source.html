41 void DeviceAddress::convertFromString(const char* str)
42 {
44  trace << "Parsing address " << str;
45  const char* pos = str;
46  constexpr unsigned int tmpSize = 20;
47  char tmp[tmpSize];
48  for (int i=0; i<size; i++)
49  {
50  unsigned int cursor = 0;
51 
52  while (*pos != '\0' && *pos != '.')
53  tmp[cursor++] = *pos++;
54 
55  if (cursor == tmpSize)
56  {
57  error << "Parsing failed: too long line";
58  return;
59  }
60 
61  if (pos == '\0' && i != size-1)
62  {
63  error << "Parsing failed: inconsistent address";
64  return;
65  }
66 
67  tmp[cursor] = '\0';
68 
69  address[i] = atoi(tmp);
70  pos++;
71  }
72  trace << "Parsed: " << address[0] << "-" << address[1] << "-" << address[2];
73 }

150 PackageId NetworkLayer::send(
151  DeviceAddress target,
152  uint8_t* data,
153  uint16_t size,
154  bool waitForAck,
155  PackageSendingDoneCallback doneCallback,
156  PackageTimings timings
157 )
158 {
159  m_stager.stage("send()");
160  if (size > Package::payloadLength)
161  {
162  error << "Error: too long payload!";
163  return 0;
164  }
165  PackageDetails details;
166  details.packageId = generatePackageId();
167 
168  if (waitForAck)
169  {
170  Time time = systemClock->getTime();
171  details.needAck = 1;
172  ScopedLock<Mutex> lock(m_packagesQueueMutex);
173  m_stager.stage("send(): +ack lock");
174  m_packages[details.packageId] = WaitingPackage();
175  WaitingPackage& waitingPackage = m_packages[details.packageId];
176  waitingPackage.wasCreated = time;
177  waitingPackage.nextTransmission = time;
178  waitingPackage.isBroadcast = broadcast.isBroadcast(target);
179 
180  waitingPackage.timings = timings;
181 
182  waitingPackage.callback = doneCallback;
183  waitingPackage.package.sender = *m_selfAddress;
184  waitingPackage.package.target = target;
185  waitingPackage.package.details = details;
186  memcpy(waitingPackage.package.payload, data, size);
188  if (size<Package::payloadLength)
189  {
190  memset(&(waitingPackage.package.payload[size]), 0, Package::payloadLength-size);
191  }
192  //trace << "Ack-using package queued";
193  m_stager.stage("send(): +ack queued");
194  return details.packageId;
195  } else {
196  ScopedLock<Mutex> lock(m_packagesQueueMutex);
197  m_stager.stage("send(): -ack lock");
198  m_packagesNoAck.push_back(Package());
199  m_packagesNoAck.back().sender = *m_selfAddress;
200  m_packagesNoAck.back().target = target;
201  m_packagesNoAck.back().details = details;
202  memcpy(m_packagesNoAck.back().payload, data, size);
203  if (size<Package::payloadLength)
204  {
205  memset(m_packagesNoAck.back().payload+size, 0, Package::payloadLength-size); //< fixed third argument
206  }
207  m_stager.stage("send(): -ack queued");
208  //trace << "No-ack package queued";
209  return 0;
210  }
211 }

220 void NetworkLayer::RXCallback(uint8_t channel, uint8_t* data)
221 {
222  UNUSED_ARG(channel);
223  m_stager.stage("RXCallback()");
224  Package received;
225  memcpy(&received, data, sizeof(Package));
226 
227  temproraryProhibitTransmission(8000);
228 
229  // Skipping packages for other devices
230  //received.target.print();
231  if (received.target != *m_selfAddress && !isBroadcast(received.target))
232  return;
233 
234  if (trace.isEnabled() && radio.isEnabled())
235  {
236  trace << "<== incoming package: ";
237  printHex((const uint8_t*) &received, sizeof(received));
238  }
239  // Dispatching if this is acknoledgement
240  AckPayload *ackDispatcher = reinterpret_cast<AckPayload *>(received.payload);
241  if (ackDispatcher->isAck())
242  {
243  radio << "<== Ack for " << ackDispatcher->packageId << " from " << ADDRESS_TO_STREAM(received.sender);
244  ScopedLock<Mutex> lock(m_packagesQueueMutex);
245  auto it = m_packages.find(ackDispatcher->packageId);
246  if (it == m_packages.end())
247  {
248  radio << "?=? No package with id " << ackDispatcher->packageId << " waiting ack";
249  // No packages waiting this ack
250  return;
251  }
252  radio << "|=| Ack for " << ackDispatcher->packageId << " accepted";
253  if (!it->second.isBroadcast)
254  {
255  PackageSendingDoneCallback callback = it->second.callback;
256  m_packages.erase(it);
257  if (callback)
258  callback(ackDispatcher->packageId, true);
259  } else {
260  radio << "|=| but package " << ackDispatcher->packageId << " is broadcast, so don\'t delete from queue";
261  }
262  return;
263  }
264 
265  //printf("Received package with id=%u\n", received.idAndTTL.packageId);
266  // Generating acknledgement
267  if (received.details.needAck == 1)
268  {
269  // Forming payload for ack package
270  AckPayload ackPayload;
271  ackPayload.packageId = received.details.packageId;
272  // Forming ack package
273  Package ack;
274  ack.target = received.sender;
275  ack.sender = *m_selfAddress;
276  ack.details.packageId = generatePackageId();
277  memcpy(&ack.payload, &ackPayload, sizeof(ackPayload));
278  memset(ack.payload+sizeof(ackPayload), 0, ack.payloadLength - sizeof(ackPayload));
279  ScopedLock<Mutex> lock(m_packagesQueueMutex);
280  // Adding ack package to list for sending
281  m_packagesNoAck.push_back(ack);
282  }
283 
284  if (checkIfIdStoredAndStore(received.details.packageId))
285  {
286  radio << "?=? Package with id " << received.details.packageId << " repetition detected";
287  } else {
288  // Putting received package to list
289  m_incoming.push_back(received);
290  }
291 }

345 void NetworkLayer::sendNext()
346 {
347  m_stager.stage("sendNext()");
348  if (!isTranslationAllowed())
349  {
350  m_stager.stage("sendNext(): Transmission disallowed");
351  return;
352  }
353  ScopedLock<Mutex> lock(m_packagesQueueMutex);
354  // First, sending packages without response
355  if (!m_packagesNoAck.empty())
356  {
357  m_stager.stage("sendNext(): sending no ack");
358  // If it is ack (only for output, may be removed by #ifdef DEBUG later)
359  AckPayload *ackDispatcher = reinterpret_cast<AckPayload *>(m_packagesNoAck.front().payload);
360  if (ackDispatcher->isAck())
361  {
362  radio << "==> Ack to " << ackDispatcher->packageId << " for " << ADDRESS_TO_STREAM(m_packagesNoAck.front().target) << " ";
363  } else {
364  radio << "==> No-ack package " << m_packagesNoAck.front().details.packageId << " for " << ADDRESS_TO_STREAM(m_packagesNoAck.front().target) << " ";
365  }
366 
367  printAndSend(m_packagesNoAck.front());
368 
369  m_packagesNoAck.pop_front();
370  isSendingNow = true;
371  // Tell that now we are sending no-response package
372  currentlySendingPackageId = 0;
373  return;
374  }
375 
376  Time time = systemClock->getTime();
377  for (auto it=m_packages.begin(); it!=m_packages.end(); )
378  // ++it in the end of cycle body for case when we removing outdated packages
379  {
380  // If timeout
381  if (!it->second.timings.infiniteResend && time - it->second.wasCreated > it->second.timings.timeout)
382  {
383  PackageSendingDoneCallback callback = it->second.callback;
384  uint16_t timeoutedPackageId = it->second.package.details.packageId;
385  radio << "==| Package " << timeoutedPackageId << " timeouted";
386  m_packages.erase(it++); // We can increment iterator
387  if (callback)
388  callback(timeoutedPackageId, false);
389  continue; // we alredy did ++it so go to next iteration
390  }
391  // If it is time to (re)send package
392  if (it->second.nextTransmission < time)
393  {
394  m_stager.stage("sendNext(): sending with ack");
395  radio << "==> Need-ack package " << it->second.package.details.packageId << " for " << ADDRESS_TO_STREAM(it->second.package.target);
396  //radio << "Transmitting " << it->second.package.details.packageId;
397  currentlySendingPackageId = it->first;
398  isSendingNow = true;
399  it->second.nextTransmission = time + it->second.timings.resendTime + Random::random(it->second.timings.resendTimeDelta);
400 
401  printAndSend(it->second.package);
402 
403  return;
404  }
405  ++it; // Now can increment
406  }
407 }
