176 void NRF24L01Manager::init(
177  IIOPin* chipEnablePin,
178  IIOPin* chipSelectPin,
179  IIOPin* IRQPin,
180  uint8_t SPIIndex,
181  bool useInterrupts,
182  uint8_t radioChannel
183 )
184 {
185  debug << "Radio module initialization...\n";
186  m_chipEnablePin = chipEnablePin;
187  m_chipSelectPin = chipSelectPin;
188  m_IRQPin = IRQPin;
189  m_RFChannel = radioChannel;
190  m_spi = SPIs->getSPI(SPIIndex);
191  m_spi->init(ISPIManager::BaudRatePrescaler32, chipSelectPin);
192 
194  // Chip enable line init
195  m_chipEnablePin->switchToOutput();
196  chipEnableOff();
197 
199  // Chip select SPI line init
200  //m_chipSelectPin->switchToOutput();
201  chipDeselect();
202 
203  // IRQ line input
204  m_IRQPin->switchToInput();
205 
207  // Configuring nrf24l01
208  // Global settings
209  trace << "setAdressWidth(AW_5BYTES);";
210  setAdressWidth(AW_5BYTES);
211  trace << "etRFChannel(m_RFChannel);";
212  setRFChannel(m_RFChannel);
213  //setupRetransmission(0, 0);
214 
215  // Reading addresses to local variables
216  trace << "readRXAdresses";
217  readRXAdresses();
218  trace << "readTXAdress";
219  readTXAdress();
220 
221  // Setting up pipe 0
222  trace << "setAutoACK";
223  setAutoACK(0, DISABLE_OPTION);
224  trace << "enablePipe";
225  enablePipe(0, ENABLE_OPTION);
226  trace << "setRXPayloadLength";
227  setRXPayloadLength(0, payloadSize);
228 
229  trace << "setRXPayloadLength";
230  setRFSettings(BR_1MBIT, PW_0DB, LNA_ENABLE);
231  trace << "setupRetransmission";
232  setupRetransmission(2, 15);
233  trace << "switchToRX";
234  switchToRX();
235 
236  m_useInterrupts = useInterrupts;
237  if (m_useInterrupts)
238  {
239  IRQPin->setExtiCallback(std::bind(&NRF24L01Manager::extiHandler, this, std::placeholders::_1), false);
240  IRQPin->enableExti(true);
241  if (false == m_IRQPin->state())
242  m_needInterrogation = true;
243  else
244  m_needInterrogation = false;
245  }
246 
247  if (m_TXAdress[0] != 0xe7)
248  {
249  error << "Radio module seems to be not connected! Details:\n";
250  printStatus();
251  }
252 }

406 void NRF24L01Manager::setConfig(unsigned char interruptionsMask,
407  unsigned char enableCRC,
408  unsigned char CRC2bytes,
409  unsigned char powerUP,
410  unsigned char isRecieving)
411 {
412  m_config =
413  (interruptionsMask << NRF_REGF_MASK_MAX_RT)
414  | (enableCRC << NRF_REGF_EN_CRC)
415  | (CRC2bytes << NRF_REGF_CRCO)
416  | (powerUP << NRF_REGF_PWR_UP)
417  | (isRecieving << NRF_REGF_PRIM_RX);
418  //printf("Config: %u\n", m_config);
419  writeReg(NRF_REG_CONFIG, 1, &m_config);
420 }

548  break;
