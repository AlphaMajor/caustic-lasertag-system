63 void MainStateSaver::saveState()
64 {
65  info << "Saving state";
66 #ifdef TWO_FILES_STATE_SAVER
67  trace << "Variant = " << m_current;
68  FRESULT res = FR_OK;
69  // Creating lock file
70  res = f_open(&m_fil, m_fileLock[m_current].c_str(), FA_CREATE_ALWAYS | FA_WRITE);
71  if (res != FR_OK)
72  {
73  error << "Cannot create lock file: " << parseFRESULT(res);
74  return;
75  }
76 
77  f_close(&m_fil);
78 
79  // Deleting old state file
80  f_unlink(m_file[m_current].c_str());
81  // Creating new state file
82  res = f_open(&m_fil, m_file[m_current].c_str(), FA_CREATE_ALWAYS | FA_WRITE);
83  if (res != FR_OK)
84  {
85  error << "Cannot create state file: " << parseFRESULT(res);
86  return;
87  }
88  // Putting data to state file
89  RCSPMultiStream stream;
90  for (OperationCode code : m_codes)
91  stream.addValue(code);
92  stream.writeToFile(&m_fil);
93  f_close(&m_fil);
94 
95  // Removing lock file
96  f_unlink(m_fileLock[m_current].c_str());
97  // Deleting old .current file
98  f_unlink(m_fileCurrent[m_next].c_str());
99  // Creating new .current file
100  res = f_open(&m_fil, m_fileCurrent[m_current].c_str(), FA_CREATE_ALWAYS | FA_WRITE);
101  if (res != FR_OK)
102  {
103  error << "Cannot create .current file: " << parseFRESULT(res);
104  return;
105  }
106  f_close(&m_fil);
107 
108  std::swap(m_current, m_next);
109 #else
110  FRESULT res = FR_OK;
111  // Creating lock file
112  res = f_open(&m_fil, m_filename.c_str(), FA_CREATE_ALWAYS | FA_WRITE);
113  if (res != FR_OK)
114  {
115  error << "Cannot create lock file: " << parseFRESULT(res);
116  return;
117  }
118  RCSPMultiStream stream;
119  for (OperationCode code : m_codes)
120  stream.addValue(code);
121  stream.writeToFile(&m_fil);
122  f_close(&m_fil);
123 #endif
124 }

126 bool MainStateSaver::tryRestore(uint8_t variant)
127 {
128 #ifdef TWO_FILES_STATE_SAVER
129  // Check if we have not deleted lock file
130  if (f_stat(m_fileLock[variant].c_str(), nullptr) == FR_OK)
131  {
132  warning << "Lock file detected, clearing\n";
133  f_unlink(m_fileLock[m_current].c_str());
134  f_unlink(m_file[m_current].c_str());
135  return false;
136  }
137  // Opening file
138  if (!f_stat(m_file[variant].c_str(), nullptr) == FR_OK)
139  {
140  warning << m_file[variant].c_str() << " does not exists\n";
141  return false;
142  }
143 
144  FRESULT res = FR_OK;
145  res = f_open(&m_fil, m_file[variant].c_str(), FA_READ);
146  if (res != FR_OK)
147  {
148  error << "Cannot open state file!\n";
149  return false;
150  }
151  uint8_t* buffer = new uint8_t[RCSPStream::defaultLength];
152  memset(buffer, 0, RCSPStream::defaultLength);
153  UINT readed = 0;
154 
155  for(;;) {
156  res = f_read(&m_fil, buffer, RCSPStream::defaultLength, &readed);
157  if (res != FR_OK)
158  {
159  error << "Error while reading state file: " << parseFRESULT(res);
160  return false;
161  }
162  if (readed != 0)
163  {
164  //trace << "Dispatching restored parameters chunk...";
165  RCSPAggregator::instance().dispatchStream(buffer, readed);
166  }
167  else
168  break;
169  }
170 
171  f_close(&m_fil);
172  return true;
173 
174 #else
175  return false;
176 #endif
177 }

179 bool MainStateSaver::tryRestore()
180 {
181 #ifdef TWO_FILES_STATE_SAVER
182  if (f_stat(m_fileCurrent[0].c_str(), nullptr) == FR_OK)
183  {
184  // If .current file points to 0 we first try to restore 0, than 1
185  return tryRestore(0) ? true : tryRestore(1);
186  }
187  // If .current file points to 1 we first try to restore 1, than 0
188  return tryRestore(1) ? true : tryRestore(0);
189 #else
190  FRESULT res = FR_OK;
191  res = f_open(&m_fil, m_filename.c_str(), FA_READ);
192  if (res != FR_OK)
193  {
194  error << "Cannot open state file!\n";
195  return false;
196  }
197  uint8_t* buffer = new uint8_t[RCSPStream::defaultLength];
198  memset(buffer, 0, RCSPStream::defaultLength);
199  UINT readed = 0;
200 
201  for(;;) {
202  res = f_read(&m_fil, buffer, RCSPStream::defaultLength, &readed);
203  if (res == FR_OK && readed != 0)
204  {
205  //trace << "Dispatching restored parameters chunk...\n";
206  RCSPAggregator::instance().dispatchStream(buffer, readed);
207  }
208  else
209  break;
210  }
211 
212  f_close(&m_fil);
213 #endif
214 }
